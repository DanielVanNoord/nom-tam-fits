32  import java.io.IOException;
33  import nom.tam.util.*;
34  import java.util.Iterator;

36  /**
37   * FITS ASCII table header/data unit
38   */
39  public class AsciiTableHDU extends TableHDU {
40  
41      /** Just a copy of myData with the correct type */
42      AsciiTable data;
43  
44      /**
45       * The standard column stems for an ASCII table. Note that TBCOL is not
46       * included here -- it needs to be handled specially since it does not
47       * simply shift.
48       */
49      private String[] keyStems = {
50          "TFORM",
51          "TZERO",
52          "TNULL",
53          "TTYPE",
54          "TUNIT"
55      };
56  
57      /**
58       * Create an ascii table header/data unit.
59       * 
60       * @param h
61       *            the template specifying the ascii table.
62       * @param d
63       *            the FITS data structure containing the table data.
64       * @exception FitsException
65       *                if there was a problem with the header.
66       */
67      public AsciiTableHDU(Header h, Data d) {
68          super((TableData) d);
69          myHeader = h;
70          data = (AsciiTable) d;
71          myData = d;
72      }
73  
74      /**
75       * Check that this is a valid ascii table header.
76       * 
77       * @param header
78       *            to validate.
79       * @return <CODE>true</CODE> if this is an ascii table header.
80       */
81      public static boolean isHeader(Header header) {
82          return header.getStringValue("XTENSION").trim().equals("TABLE");
83      }
84  
85      /**
86       * Check that this HDU has a valid header.
87       * 
88       * @return <CODE>true</CODE> if this HDU has a valid header.
89       */
90      public boolean isHeader() {
91          return isHeader(myHeader);
92      }
93  
94      /**
95       * Check if this data is usable as an ASCII table.
96       */
97      public static boolean isData(Object o) {
98  
99          if (o instanceof Object[]) {
100             Object[] oo = (Object[]) o;
101             for (int i = 0; i < oo.length; i += 1) {
102                 if (oo[i] instanceof String[] || oo[i] instanceof int[] || oo[i] instanceof long[] || oo[i] instanceof float[] || oo[i] instanceof double[]) {
103                     continue;
104                 }
105                 return false;
106             }
107             return true;
108         } else {
109             return false;
110         }
111     }
112 
113     /**
114      * Create a Data object to correspond to the header description.
115      * 
116      * @return An unfilled Data object which can be used to read in the data for
117      *         this HDU.
118      * @exception FitsException
119      *                if the Data object could not be created from this HDU's
120      *                Header
121      */
122     public static Data manufactureData(Header hdr) throws FitsException {
123         return new AsciiTable(hdr);
124     }
125 
126     /**
127      * Create an empty data structure corresponding to the input header.
128      */
129     public Data manufactureData() throws FitsException {
130         return manufactureData(myHeader);
131     }
132 
133     /** Create a header to match the input data. */
134     public static Header manufactureHeader(Data d) throws FitsException {
135         Header hdr = new Header();
136         d.fillHeader(hdr);
137         Iterator iter = hdr.iterator();
138         return hdr;
139     }
140 
141     /**
142      * Create a ASCII table data structure from an array of objects representing
143      * the columns.
144      */
145     public static Data encapsulate(Object o) throws FitsException {
146 
147         Object[] oo = (Object[]) o;
148         AsciiTable d = new AsciiTable();
149         for (int i = 0; i < oo.length; i += 1) {
150             d.addColumn(oo[i]);
151         }
152         return d;
153     }
154 
155     /**
156      * Skip the ASCII table and throw an exception.
157      * 
158      * @param stream
159      *            the stream from which the data is read.
160      */
161     public void readData(ArrayDataInput stream) throws FitsException {
162         myData.read(stream);
163     }
164 
165     /**
166      * Mark an entry as null.
167      */
168     public void setNull(int row, int col, boolean flag) {
169 
170         if (flag) {
171             String nullStr = myHeader.getStringValue("TNULL" + (col + 1));
172             if (nullStr == null) {
173                 setNullString(col, "NULL");
174             }
175         }
176         data.setNull(row, col, flag);
177     }
178 
179     /** See if an element is null */
180     public boolean isNull(int row, int col) {
181         return data.isNull(row, col);
182     }
183 
184     /** Set the null string for a column */
185     public void setNullString(int col, String newNull) {
186         myHeader.positionAfterIndex("TBCOL", col + 1);
187         try {
188             myHeader.addValue("TNULL" + (col + 1), newNull, "ntf::asciitablehdu:tnullN:1");
189         } catch (HeaderCardException e) {
190             System.err.println("Impossible exception in setNullString" + e);
191         }
192         data.setNullString(col, newNull);
193     }
194 
195     /** Add a column */
196     public int addColumn(Object newCol) throws FitsException {
197 
198         data.addColumn(newCol);
199 
200         // Move the iterator to point after all the data describing
201         // the previous column.
202 
203         Cursor iter = myHeader.positionAfterIndex("TBCOL", data.getNCols());
204 
205         int rowlen = data.addColInfo(getNCols(), iter);
206         int oldRowlen = myHeader.getIntValue("NAXIS1");
207         myHeader.setNaxis(1, rowlen + oldRowlen);
208 
209         int oldTfields = myHeader.getIntValue("TFIELDS");
210         try {
211             myHeader.addValue("TFIELDS", oldTfields + 1, "ntf::asciitablehdu:tfields:1");
212         } catch (Exception e) {
213             System.err.println("Impossible exception at addColumn:" + e);
214         }
215         return getNCols();
216     }
217 
218     /**
219      * Print a little information about the data set.
220      */
221     public void info() {
222         System.out.println("ASCII Table:");
223         System.out.println("  Header:");
224         System.out.println("    Number of fields:" + myHeader.getIntValue("TFIELDS"));
225         System.out.println("    Number of rows:  " + myHeader.getIntValue("NAXIS2"));
226         System.out.println("    Length of row:   " + myHeader.getIntValue("NAXIS1"));
227         System.out.println("  Data:");
228         Object[] data = (Object[]) getKernel();
229         for (int i = 0; i < getNCols(); i += 1) {
230             System.out.println("      " + i + ":" + ArrayFuncs.arrayDescription(data[i]));
231         }
232     }
233 
234     /**
235      * Return the FITS data structure associated with this HDU.
236      */
237     public Data getData() {
238         return data;
239     }
240 
241     /**
242      * Return the keyword column stems for an ASCII table.
243      */
244     public String[] columnKeyStems() {
245         return keyStems;
246     }
