34  import java.util.Iterator;
35  
36  import nom.tam.util.ArrayDataInput;
37  import nom.tam.util.ArrayFuncs;
38  import nom.tam.util.Cursor;

40  /**
41   * FITS ASCII table header/data unit
42   */
43  public class AsciiTableHDU extends TableHDU {
44  
45      /** Just a copy of myData with the correct type */
46      AsciiTable data;
47  
48      /**
49       * The standard column stems for an ASCII table. Note that TBCOL is not
50       * included here -- it needs to be handled specially since it does not
51       * simply shift.
52       */
53      private String[] keyStems = {
54          "TFORM",
55          "TZERO",
56          "TNULL",
57          "TTYPE",
58          "TUNIT"
59      };
60  
61      /**
62       * Create an ascii table header/data unit.
63       * 
64       * @param h
65       *            the template specifying the ascii table.
66       * @param d
67       *            the FITS data structure containing the table data.
68       * @exception FitsException
69       *                if there was a problem with the header.
70       */
71      public AsciiTableHDU(Header h, Data d) {
72          super((TableData) d);
73          myHeader = h;
74          data = (AsciiTable) d;
75          myData = d;
76      }
77  
78      /**
79       * Check that this is a valid ascii table header.
80       * 
81       * @param header
82       *            to validate.
83       * @return <CODE>true</CODE> if this is an ascii table header.
84       */
85      public static boolean isHeader(Header header) {
86          return header.getStringValue("XTENSION").trim().equals("TABLE");
87      }
88  
89      /**
90       * Check that this HDU has a valid header.
91       * 
92       * @return <CODE>true</CODE> if this HDU has a valid header.
93       */
94      public boolean isHeader() {
95          return isHeader(myHeader);
96      }
97  
98      /**
99       * Check if this data is usable as an ASCII table.
100      */
101     public static boolean isData(Object o) {
102 
103         if (o instanceof Object[]) {
104             Object[] oo = (Object[]) o;
105             for (int i = 0; i < oo.length; i += 1) {
106                 if (oo[i] instanceof String[] || oo[i] instanceof int[] || oo[i] instanceof long[] || oo[i] instanceof float[] || oo[i] instanceof double[]) {
107                     continue;
108                 }
109                 return false;
110             }
111             return true;
112         } else {
113             return false;
114         }
115     }
116 
117     /**
118      * Create a Data object to correspond to the header description.
119      * 
120      * @return An unfilled Data object which can be used to read in the data for
121      *         this HDU.
122      * @exception FitsException
123      *                if the Data object could not be created from this HDU's
124      *                Header
125      */
126     public static Data manufactureData(Header hdr) throws FitsException {
127         return new AsciiTable(hdr);
128     }
129 
130     /**
131      * Create an empty data structure corresponding to the input header.
132      */
133     public Data manufactureData() throws FitsException {
134         return manufactureData(myHeader);
135     }
136 
137     /** Create a header to match the input data. */
138     public static Header manufactureHeader(Data d) throws FitsException {
139         Header hdr = new Header();
140         d.fillHeader(hdr);
141         Iterator iter = hdr.iterator();
142         return hdr;
143     }
144 
145     /**
146      * Create a ASCII table data structure from an array of objects representing
147      * the columns.
148      */
149     public static Data encapsulate(Object o) throws FitsException {
150 
151         Object[] oo = (Object[]) o;
152         AsciiTable d = new AsciiTable();
153         for (int i = 0; i < oo.length; i += 1) {
154             d.addColumn(oo[i]);
155         }
156         return d;
157     }
158 
159     /**
160      * Skip the ASCII table and throw an exception.
161      * 
162      * @param stream
163      *            the stream from which the data is read.
164      */
165     public void readData(ArrayDataInput stream) throws FitsException {
166         myData.read(stream);
167     }
168 
169     /**
170      * Mark an entry as null.
171      */
172     public void setNull(int row, int col, boolean flag) {
173 
174         if (flag) {
175             String nullStr = myHeader.getStringValue("TNULL" + (col + 1));
176             if (nullStr == null) {
177                 setNullString(col, "NULL");
178             }
179         }
180         data.setNull(row, col, flag);
181     }
182 
183     /** See if an element is null */
184     public boolean isNull(int row, int col) {
185         return data.isNull(row, col);
186     }
187 
188     /** Set the null string for a column */
189     public void setNullString(int col, String newNull) {
190         myHeader.positionAfterIndex("TBCOL", col + 1);
191         try {
192             myHeader.addValue("TNULL" + (col + 1), newNull, "ntf::asciitablehdu:tnullN:1");
193         } catch (HeaderCardException e) {
194             System.err.println("Impossible exception in setNullString" + e);
195         }
196         data.setNullString(col, newNull);
197     }
198 
199     /** Add a column */
200     public int addColumn(Object newCol) throws FitsException {
201 
202         data.addColumn(newCol);
203 
204         // Move the iterator to point after all the data describing
205         // the previous column.
206 
207         Cursor iter = myHeader.positionAfterIndex("TBCOL", data.getNCols());
208 
209         int rowlen = data.addColInfo(getNCols(), iter);
210         int oldRowlen = myHeader.getIntValue("NAXIS1");
211         myHeader.setNaxis(1, rowlen + oldRowlen);
212 
213         int oldTfields = myHeader.getIntValue("TFIELDS");
214         try {
215             myHeader.addValue("TFIELDS", oldTfields + 1, "ntf::asciitablehdu:tfields:1");
216         } catch (Exception e) {
217             System.err.println("Impossible exception at addColumn:" + e);
218         }
219         return getNCols();
220     }
221 
222     /**
223      * Print a little information about the data set.
224      */
225     public void info() {
226         System.out.println("ASCII Table:");
227         System.out.println("  Header:");
228         System.out.println("    Number of fields:" + myHeader.getIntValue("TFIELDS"));
229         System.out.println("    Number of rows:  " + myHeader.getIntValue("NAXIS2"));
230         System.out.println("    Length of row:   " + myHeader.getIntValue("NAXIS1"));
231         System.out.println("  Data:");
232         Object[] data = (Object[]) getKernel();
233         for (int i = 0; i < getNCols(); i += 1) {
234             System.out.println("      " + i + ":" + ArrayFuncs.arrayDescription(data[i]));
235         }
236     }
237 
238     /**
239      * Return the FITS data structure associated with this HDU.
240      */
241     public Data getData() {
242         return data;
243     }
244 
245     /**
246      * Return the keyword column stems for an ASCII table.
247      */
248     public String[] columnKeyStems() {
249         return keyStems;
250     }
