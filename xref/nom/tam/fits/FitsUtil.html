63      /** Reposition a random access stream to a requested offset */
64      public static void reposition(Object o, long offset) throws FitsException {
65  
66          if (o == null) {
67              throw new FitsException("Attempt to reposition null stream");
68          }
69          if (!(o instanceof RandomAccess) || offset < 0) {
70              throw new FitsException("Invalid attempt to reposition stream " + o + " of type " + o.getClass().getName() + " to " + offset);
71          }
72  
73          try {
74              ((RandomAccess) o).seek(offset);
75          } catch (IOException e) {
76              throw new FitsException("Unable to repostion stream " + o + " of type " + o.getClass().getName() + " to " + offset + "   Exception:" + e);
77          }
78      }

80      /** Find out where we are in a random access file */
81      public static long findOffset(Object o) {
82  
83          if (o instanceof RandomAccess) {
84              return ((RandomAccess) o).getFilePointer();
85          } else {
86              return -1;
87          }
88      }

91      public static int padding(int size) {

95      public static int padding(long size) {
96  
97          int mod = (int) (size % 2880);
98          if (mod > 0) {
99              mod = 2880 - mod;
100         }
101         return mod;
102     }

113     /**
114      * This method decompresses a compressed input stream. The decompression
115      * method is selected automatically based upon the first two bytes read.
116      * 
117      * @param compressed
118      *            The compressed input stram
119      * @return A stream which wraps the input stream and decompresses it. If the
120      *         input stream is not compressed, a pushback input stream wrapping
121      *         the original stream is returned.
122      */
123     static InputStream decompress(InputStream compressed) throws FitsException {
124 
125         PushbackInputStream pb = new PushbackInputStream(compressed, 2);
126 
127         int mag1 = -1;
128         int mag2 = -1;
129 
130         try {
131             mag1 = pb.read();
132             mag2 = pb.read();
133 
134             if (mag1 == 0x1f && mag2 == 0x8b) {
135                 // Push the data back into the stream
136                 pb.unread(mag2);
137                 pb.unread(mag1);
138                 return new GZIPInputStream(pb);
139             } else if (mag1 == 0x1f && mag2 == 0x9d) {
140                 // Push the data back into the stream
141                 pb.unread(mag2);
142                 pb.unread(mag1);
143                 return compressInputStream(pb);
144             } else if (mag1 == 'B' && mag2 == 'Z') {
145                 if (System.getenv("BZIP_DECOMPRESSOR") != null) {
146                     pb.unread(mag2);
147                     pb.unread(mag1);
148                     return bunzipper(pb);
149                 }
150                 pb.unread(mag2);
151                 pb.unread(mag1);
152                 return new BZip2CompressorInputStream(pb);
153 
154             } else {
155                 // Push the data back into the stream
156                 pb.unread(mag2);
157                 pb.unread(mag1);
158                 return pb;
159             }
160 
161         } catch (IOException e) {
162             // This is probably a prelude to failure...
163             throw new FitsException("Unable to analyze input stream");
164         }
165     }

167     static InputStream compressInputStream(final InputStream compressed) throws FitsException {
168         try {
169             Process proc = new ProcessBuilder("uncompress", "-c").start();
170 
171             // This is the input to the process -- but
172             // an output from here.
173             final OutputStream input = proc.getOutputStream();
174 
175             // Now copy everything in a separate thread.
176             Thread copier = new Thread(new Runnable() {
177 
178                 public void run() {
179                     try {
180                         byte[] buffer = new byte[8192];
181                         int len;
182                         while ((len = compressed.read(buffer, 0, buffer.length)) > 0) {
183                             input.write(buffer, 0, len);
184                         }
185                         compressed.close();
186                         input.close();
187                     } catch (IOException e) {
188                         return;
189                     }
190                 }
191             });
192             copier.start();
193             return proc.getInputStream();
194         } catch (Exception e) {
195             throw new FitsException("Unable to read .Z compressed stream.\nIs `uncompress' in the path?\n:" + e);
196         }
197     }

199     /** Is a file compressed? */
200     public static boolean isCompressed(File test) {
201         InputStream fis = null;
202         try {
203             if (test.exists()) {
204                 fis = new FileInputStream(test);
205                 int mag1 = fis.read();
206                 int mag2 = fis.read();
207                 fis.close();
208                 if (mag1 == 0x1f && (mag2 == 0x8b || mag2 == 0x9d)) {
209                     return true;
210                 } else if (mag1 == 'B' && mag2 == 'Z') {
211                     return true;
212                 } else {
213                     return false;
214                 }
215             }
216 
217         } catch (IOException e) {
218             // This is probably a prelude to failure...
219             return false;
220 
221         } finally {
222             if (fis != null) {
223                 try {
224                     fis.close();
225                 } catch (IOException e) {
226                 }
227             }
228         }
229         return false;
230     }

232     /**
233      * Check if a file seems to be compressed.
234      */
235     public static boolean isCompressed(String filename) {
236         if (filename == null) {
237             return false;
238         }
239         FileInputStream fis = null;
240         File test = new File(filename);
241         if (test.exists()) {
242             return isCompressed(test);
243         }
244 
245         int len = filename.length();
246         return len > 2 && (filename.substring(len - 3).equalsIgnoreCase(".gz") || filename.substring(len - 2).equals(".Z"));
247     }

249     /**
250      * Get the maximum length of a String in a String array.
251      */
252     public static int maxLength(String[] o) throws FitsException {
253 
254         int max = 0;
255         for (int i = 0; i < o.length; i += 1) {
256             if (o[i] != null && o[i].length() > max) {
257                 max = o[i].length();
258             }
259         }
260         return max;
261     }

263     /** Copy an array of Strings to bytes. */
264     public static byte[] stringsToByteArray(String[] o, int maxLen) {
265         byte[] res = new byte[o.length * maxLen];
266         for (int i = 0; i < o.length; i += 1) {
267             byte[] bstr = null;
268             if (o[i] == null) {
269                 bstr = new byte[0];
270             } else {
271                 bstr = AsciiFuncs.getBytes(o[i]);
272             }
273             int cnt = bstr.length;
274             if (cnt > maxLen) {
275                 cnt = maxLen;
276             }
277             System.arraycopy(bstr, 0, res, i * maxLen, cnt);
278             for (int j = cnt; j < maxLen; j += 1) {
279                 res[i * maxLen + j] = (byte) ' ';
280             }
281         }
282         return res;
283     }

285     /** Convert bytes to Strings */
286     public static String[] byteArrayToStrings(byte[] o, int maxLen) {
287         boolean checking = FitsFactory.getCheckAsciiStrings();
288 
289         // Note that if a String in a binary table contains an internal 0,
290         // the FITS standard says that it is to be considered as terminating
291         // the string at that point, so that software reading the
292         // data back may not include subsequent characters.
293         // No warning of this truncation is given.
294 
295         String[] res = new String[o.length / maxLen];
296         for (int i = 0; i < res.length; i += 1) {
297 
298             int start = i * maxLen;
299             int end = start + maxLen;
300             // Pre-trim the string to avoid keeping memory
301             // hanging around. (Suggested by J.C. Segovia, ESA).
302 
303             // Note that the FITS standard does not mandate
304             // that we should be trimming the string at all, but
305             // this seems to best meet the desires of the community.
306             for (; start < end; start += 1) {
307                 if (o[start] != 32) {
308                     break; // Skip only spaces.
309                 }
310             }
311 
312             for (; end > start; end -= 1) {
313                 if (o[end - 1] != 32) {
314                     break;
315                 }
316             }
317 
318             // For FITS binary tables, 0 values are supposed
319             // to terminate strings, a la C. [They shouldn't appear in
320             // any other context.]
321             // Other non-printing ASCII characters
322             // should always be an error which we can check for
323             // if the user requests.
324 
325             // The lack of handling of null bytes was noted by Laurent Bourges.
326             boolean errFound = false;
327             for (int j = start; j < end; j += 1) {
328 
329                 if (o[j] == 0) {
330                     end = j;
331                     break;
332                 }
333                 if (checking) {
334                     if (o[j] < 32 || o[j] > 126) {
335                         errFound = true;
336                         o[j] = 32;
337                     }
338                 }
339             }
340             res[i] = AsciiFuncs.asciiString(o, start, end - start);
341             if (errFound && !wroteCheckingError) {
342                 System.err.println("Warning: Invalid ASCII character[s] detected in string:" + res[i]);
343                 System.err.println("   Converted to space[s].  Any subsequent invalid characters will be converted silently");
344                 wroteCheckingError = true;
345             }
346         }
347         return res;
348 
349     }

351     /** Convert an array of booleans to bytes */
352     static byte[] booleanToByte(boolean[] bool) {
353 
354         byte[] byt = new byte[bool.length];
355         for (int i = 0; i < bool.length; i += 1) {
356             byt[i] = bool[i] ? (byte) 'T' : (byte) 'F';
357         }
358         return byt;
359     }

361     /** Convert an array of bytes to booleans */
362     static boolean[] byteToBoolean(byte[] byt) {
363         boolean[] bool = new boolean[byt.length];
364 
365         for (int i = 0; i < byt.length; i += 1) {
366             bool[i] = (byt[i] == 'T');
367         }
368         return bool;
369     }

371     /**
372      * Get a stream to a URL accommodating possible redirections. Note that if a
373      * redirection request points to a different protocol than the original
374      * request, then the redirection is not handled automatically.
375      */
376     public static InputStream getURLStream(URL url, int level) throws IOException {
377 
378         // Hard coded....sigh
379         if (level > 5) {
380             throw new IOException("Two many levels of redirection in URL");
381         }
382         URLConnection conn = url.openConnection();
383         // Map<String,List<String>> hdrs = conn.getHeaderFields();
384         Map hdrs = conn.getHeaderFields();
385 
386         // Read through the headers and see if there is a redirection header.
387         // We loop (rather than just do a get on hdrs)
388         // since we want to match without regard to case.
389         String[] keys = (String[]) hdrs.keySet().toArray(new String[0]);
390         // for (String key: hdrs.keySet()) {
391         for (int i = 0; i < keys.length; i += 1) {
392             String key = keys[i];
393 
394             if (key != null && key.toLowerCase().equals("location")) {
395                 // String val = hdrs.get(key).get(0);
396                 String val = (String) ((List) hdrs.get(key)).get(0);
397                 if (val != null) {
398                     val = val.trim();
399                     if (val.length() > 0) {
400                         // Redirect
401                         return getURLStream(new URL(val), level + 1);
402                     }
403                 }
404             }
405         }
406         // No redirection
407         return conn.getInputStream();
408     }

410     /** Add padding to an output stream. */
411     public static void pad(ArrayDataOutput stream, long size) throws FitsException {
412         pad(stream, size, (byte) 0);
413     }

415     /** Add padding to an output stream. */
416     public static void pad(ArrayDataOutput stream, long size, byte fill) throws FitsException {
417         int len = padding(size);
418         if (len > 0) {
419             byte[] buf = new byte[len];
420             for (int i = 0; i < len; i += 1) {
421                 buf[i] = fill;
422             }
423             try {
424                 stream.write(buf);
425                 stream.flush();
426             } catch (Exception e) {
427                 throw new FitsException("Unable to write padding", e);
428             }
429         }
430     }

432     static InputStream bunzipper(final InputStream pb) throws FitsException {
433         String cmd = System.getenv("BZIP_DECOMPRESSOR");
434         // Allow the user to have already specified the - option.
435         if (cmd.indexOf(" -") < 0) {
436             cmd += " -";
437         }
438         final OutputStream out;
439         String[] flds = cmd.split(" +");
440         Thread t;
441         Process p;
442         try {
443             p = new ProcessBuilder(flds).start();
444             out = p.getOutputStream();
445 
446             t = new Thread(new Runnable() {
447 
448                 public void run() {
449                     try {
450                         byte[] buf = new byte[16384];
451                         int len;
452                         long total = 0;
453                         while ((len = pb.read(buf)) > 0) {
454                             try {
455                                 out.write(buf, 0, len);
456                             } catch (Exception e) {
457                                 // Skip this. It can happen when we
458                                 // stop reading the compressed file in mid
459                                 // stream.
460                                 break;
461                             }
462                             total += len;
463                         }
464                         pb.close();
465                         out.close();
466 
467                     } catch (IOException e) {
468                         throw new Error("Error reading BZIP compression using: " + System.getenv("BZIP_DECOMPRESSOR"), e);
469                     }
470                 }
471             });
472 
473         } catch (Exception e) {
474             throw new FitsException("Error initiating BZIP decompression: " + e);
475         }
476         t.start();
477         return new CloseIS(p.getInputStream(), pb, out);
478 
479     }
