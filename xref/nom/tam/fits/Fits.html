52  /**
53   * This class provides access to routines to allow users to read and write FITS
54   * files.
55   * <p>
56   * <p>
57   * <b> Description of the Package </b>
58   * <p>
59   * This FITS package attempts to make using FITS files easy, but does not do
60   * exhaustive error checking. Users should not assume that just because a FITS
61   * file can be read and written that it is necessarily legal FITS. These classes
62   * try to make it easy to transform between arrays of Java primitives and their
63   * FITS encodings.
64   * <ul>
65   * <li>The Fits class provides capabilities to read and write data at the HDU
66   * level, and to add and delete HDU's from the current Fits object. A large
67   * number of constructors are provided which allow users to associate the Fits
68   * object with some form of external data. This external data may be in a
69   * compressed format.
70   * <li>The HDU class is a factory class which is used to create HDUs. HDU's can
71   * be of a number of types derived from the abstract class BasicHDU. The
72   * hierarchy of HDUs is:
73   * <ul>
74   * <li>BasicHDU
75   * <ul>
76   * <li>ImageHDU
77   * <li>RandomGroupsHDU
78   * <li>TableHDU
79   * <ul>
80   * <li>BinaryTableHDU
81   * <li>AsciiTableHDU
82   * </ul>
83   * </ul>
84   * </ul>
85   * <li>The Header class provides many functions to add, delete and read header
86   * keywords in a variety of formats.
87   * <li>The HeaderCard class provides access to the structure of a FITS header
88   * card.
89   * <li>The Data class is an abstract class which provides the basic methods for
90   * reading and writing FITS data. Users will likely only be interested in the
91   * getData method which returns that actual FITS data.
92   * <li>The TableHDU class provides a large number of methods to access and
93   * modify information in tables.
94   * <li>The Column class combines the Header information and Data corresponding
95   * to a given column.
96   * </ul>
97   * 
98   * @version 1.11
99   */

470     /**
471      * Initialize the input stream. Mostly this checks to see if the stream is
472      * compressed and wraps the stream if necessary. Even if the stream is not
473      * compressed, it will likely be wrapped in a PushbackInputStream. So users
474      * should probably not supply a BufferedDataInputStream themselves, but
475      * should allow the Fits class to do the wrapping.
476      * 
477      * @param str
478      * @throws FitsException
479      */
480     protected void streamInit(InputStream str) throws FitsException {
481         str = FitsUtil.decompress(str);
482         if (str instanceof ArrayDataInput) {
483             dataStr = (ArrayDataInput) str;
484         } else {
485             // Use efficient blocking for input.
486             dataStr = new BufferedDataInputStream(str);
487         }
488     }
489 
490     /**
491      * Initialize using buffered random access. This implies that the data is
492      * uncompressed.
493      * 
494      * @param f
495      * @throws FitsException
496      */
497     protected void randomInit(File f) throws FitsException {
498 
499         String permissions = "r";
500         if (!f.exists() || !f.canRead()) {
501             throw new FitsException("Non-existent or unreadable file");
502         }
503         if (f.canWrite()) {
504             permissions += "w";
505         }
506         try {
507             dataStr = new BufferedFile(f, permissions);
508 
509             ((BufferedFile) dataStr).seek(0);
510         } catch (IOException e) {
511             throw new FitsException("Unable to open file " + f.getPath());
512         }
513     }

539      * Get a stream from the file and then use the stream initialization.

542      *            The File to be associated.
543      * @param compressed
544      *            Is the data compressed?
545      */
546     protected void fileInit(File myFile, boolean compressed) throws FitsException {
547 
548         try {
549             if (compressed) {
550                 FileInputStream str = new FileInputStream(myFile);
551                 streamInit(str);
552             } else {
553                 randomInit(myFile);
554             }
555         } catch (IOException e) {
556             throw new FitsException("Unable to create Input Stream from File: " + myFile);
557         }
558     }

576     /**
577      * Associate the FITS object with a file or URL. The string is assumed to be
578      * a URL if it begins one of the protocol strings. If the string ends in .gz
579      * it is assumed that the data is in a compressed format. All string
580      * comparisons are case insensitive.
581      * 
582      * @param filename
583      *            The name of the file or URL to be processed.
584      * @exception FitsException
585      *                Thrown if unable to find or open a file or URL from the
586      *                string given.
587      **/
588     public Fits(String filename, boolean compressed) throws FitsException {
589 
590         InputStream inp;
591 
592         if (filename == null) {
593             throw new FitsException("Null FITS Identifier String");
594         }
595 
596         int len = filename.length();
597         String lc = filename.toLowerCase();
598         try {
599             URL test = new URL(filename);
600             InputStream is = FitsUtil.getURLStream(new URL(filename), 0);
601             streamInit(is);
602             return;
603         } catch (Exception e) {
604             // Just try it as a file
605         }
606 
607         File fil = new File(filename);
608         if (fil.exists()) {
609             fileInit(fil, compressed);
610             return;
611         }
612 
613         try {
614             InputStream str = ClassLoader.getSystemClassLoader().getResourceAsStream(filename);
615             streamInit(str);
616         } catch (Exception e) {
617             //
618         }
619 
620     }

652     /**
653      * Return all HDUs for the Fits object. If the FITS file is associated with
654      * an external stream make sure that we have exhausted the stream.
655      * 
656      * @return an array of all HDUs in the Fits object. Returns null if there
657      *         are no HDUs associated with this object.
658      */
659     public BasicHDU[] read() throws FitsException {
660 
661         readToEnd();
662 
663         int size = getNumberOfHDUs();
664 
665         if (size == 0) {
666             return null;
667         }
668 
669         BasicHDU[] hdus = new BasicHDU[size];
670         hduList.copyInto(hdus);
671         return hdus;
672     }

681     public BasicHDU readHDU() throws FitsException, IOException {
682 
683         if (dataStr == null || atEOF) {
684             return null;
685         }
686 
687         if (dataStr instanceof nom.tam.util.RandomAccess && lastFileOffset > 0) {
688             FitsUtil.reposition(dataStr, lastFileOffset);
689         }
690 
691         Header hdr = Header.readHeader(dataStr);
692         if (hdr == null) {
693             atEOF = true;
694             return null;
695         }
696 
697         Data datum = hdr.makeData();
698         try {
699             datum.read(dataStr);
700         } catch (PaddingException e) {
701             e.updateHeader(hdr);
702             throw e;
703         }
704 
705         lastFileOffset = FitsUtil.findOffset(dataStr);
706         BasicHDU nextHDU = FitsFactory.HDUFactory(hdr, datum);
707 
708         hduList.addElement(nextHDU);
709         return nextHDU;
710     }

809     /**
810      * Insert a FITS object into the list of HDUs.
811      * 
812      * @param myHDU
813      *            The HDU to be inserted into the list of HDUs.
814      * @param n
815      *            The location at which the HDU is to be inserted.
816      */
817     public void insertHDU(BasicHDU myHDU, int n) throws FitsException {
818 
819         if (myHDU == null) {
820             return;
821         }
822 
823         if (n < 0 || n > getNumberOfHDUs()) {
824             throw new FitsException("Attempt to insert HDU at invalid location: " + n);
825         }
826 
827         try {
828 
829             if (n == 0) {
830 
831                 // Note that the previous initial HDU is no longer the first.
832                 // If we were to insert tables backwards from last to first,
833                 // we could get a lot of extraneous DummyHDUs but we currently
834                 // do not worry about that.
835 
836                 if (getNumberOfHDUs() > 0) {
837                     ((BasicHDU) hduList.elementAt(0)).setPrimaryHDU(false);
838                 }
839 
840                 if (myHDU.canBePrimary()) {
841                     myHDU.setPrimaryHDU(true);
842                     hduList.insertElementAt(myHDU, 0);
843                 } else {
844                     insertHDU(BasicHDU.getDummyHDU(), 0);
845                     myHDU.setPrimaryHDU(false);
846                     hduList.insertElementAt(myHDU, 1);
847                 }
848             } else {
849                 myHDU.setPrimaryHDU(false);
850                 hduList.insertElementAt(myHDU, n);
851             }
852         } catch (NoSuchElementException e) {
853             throw new FitsException("hduList inconsistency in insertHDU");
854         }
855 
856     }

859      * Delete an HDU from the HDU list.
860      * 
861      * @param n
862      *            The index of the HDU to be deleted. If n is 0 and there is
863      *            more than one HDU present, then the next HDU will be converted
864      *            from an image to primary HDU if possible. If not a dummy
865      *            header HDU will then be inserted.
866      */

870             throw new FitsException("Attempt to delete non-existent HDU:" + n);
871         }
872         try {
873             hduList.removeElementAt(n);
874             if (n == 0 && size > 1) {
875                 BasicHDU newFirst = (BasicHDU) hduList.elementAt(0);
876                 if (newFirst.canBePrimary()) {
877                     newFirst.setPrimaryHDU(true);
878                 } else {
879                     insertHDU(BasicHDU.getDummyHDU(), 0);
880                 }
881             }
882         } catch (NoSuchElementException e) {
883             throw new FitsException("Internal Error: hduList Vector Inconsitency");
884         }

888      * Write a Fits Object to an external Stream.
889      * 
890      * @param os
891      *            A DataOutput stream.
892      */
893     public void write(DataOutput os) throws FitsException {
894 
895         ArrayDataOutput obs;
896         boolean newOS = false;
897 
898         if (os instanceof ArrayDataOutput) {
899             obs = (ArrayDataOutput) os;
900         } else if (os instanceof DataOutputStream) {
901             newOS = true;
902             obs = new BufferedDataOutputStream((DataOutputStream) os);
903         } else {
904             throw new FitsException("Cannot create ArrayDataOutput from class " + os.getClass().getName());
905         }
906 
907         BasicHDU hh;
908         for (int i = 0; i < getNumberOfHDUs(); i += 1) {
909             try {
910                 hh = (BasicHDU) hduList.elementAt(i);
911                 hh.write(obs);
912             } catch (ArrayIndexOutOfBoundsException e) {
913                 e.printStackTrace();
914                 throw new FitsException("Internal Error: Vector Inconsistency" + e);
915             }
916         }
917         if (newOS) {
918             try {
919                 obs.flush();
920                 obs.close();
921             } catch (IOException e) {
922                 System.err.println("Warning: error closing FITS output stream");
923             }
924         }
925         try {
926             if (obs instanceof BufferedFile) {
927                 ((BufferedFile) obs).setLength(((BufferedFile) obs).getFilePointer());
928             }
929         } catch (IOException e) {
930             // Ignore problems...
931         }
932 
933     }

1035     /**
1036      * Add or update the CHECKSUM keyword.
1037      * 
1038      * @param hdu
1039      *            the HDU to be updated.
1040      * @throws nom.tam.fits.HeaderCardException
1041      * @author R J Mathar
1042      * @since 2005-10-05
1043      */
1044     public static void setChecksum(BasicHDU hdu) throws nom.tam.fits.HeaderCardException, nom.tam.fits.FitsException, java.io.IOException {
1045         /*
1046          * the next line with the delete is needed to avoid some unexpected
1047          * problems with non.tam.fits.Header.checkCard() which otherwise says it
1048          * expected PCOUNT and found DATE.
1049          */
1050         Header hdr = hdu.getHeader();
1051         hdr.deleteKey("CHECKSUM");
1052         /*
1053          * jThis would need org.nevec.utils.DateUtils compiled before
1054          * org.nevec.prima.fits .... final String doneAt =
1055          * DateUtils.dateToISOstring(0) ; We need to save the value of the
1056          * comment string because this is becoming part of the checksum
1057          * calculated and needs to be re-inserted again - with the same string -
1058          * when the second/final call to addValue() is made below.
1059          */
1060         final String doneAt = HeaderCommentsMap.getComment("fits:checksum:1");
1061         hdr.addValue("CHECKSUM", "0000000000000000", doneAt);
1062 
1063         /*
1064          * Convert the entire sequence of 2880 byte header cards into a byte
1065          * array. The main benefit compared to the C implementations is that we
1066          * do not need to worry about the particular byte order on machines
1067          * (Linux/VAX/MIPS vs Hp-UX, Sparc...) supposed that the correct
1068          * implementation is in the write() interface.
1069          */
1070         ByteArrayOutputStream hduByteImage = new ByteArrayOutputStream();
1071         BufferedDataOutputStream bdos = new BufferedDataOutputStream(hduByteImage);
1072 
1073         // DATASUM keyword.
1074         hdu.getData().write(bdos);
1075         bdos.flush();
1076         byte[] data = hduByteImage.toByteArray();
1077         long checksum = checksum(data);
1078         hdu.write(new BufferedDataOutputStream(hduByteImage));
1079         long csd = checksum(data);
1080         hdu.getHeader().addValue("DATASUM", "" + csd, "Checksum of data");
1081 
1082         // We already have the checsum of the data. Lets compute it for
1083         // the header.
1084         hduByteImage.reset();
1085         hdu.getHeader().write(bdos);
1086         bdos.flush();
1087         data = hduByteImage.toByteArray();
1088 
1089         long csh = checksum(data);
1090 
1091         long cshdu = csh + csd;
1092         // If we had a carry it should go into the
1093         // beginning.
1094         while ((cshdu & 0xFFFFFFFF00000000L) != 0) {
1095             cshdu = (cshdu & 0xFFFFFFFFL) + 1;
1096         }
1097         /*
1098          * This time we do not use a deleteKey() to ensure that the keyword is
1099          * replaced "in place". Note that the value of the checksum is actually
1100          * independent to a permutation of the 80-byte records within the
1101          * header.
1102          */
1103         hdr.addValue("CHECKSUM", checksumEnc(cshdu, true), doneAt);
1104     }

1120     /**
1121      * Calculate the Seaman-Pence 32-bit 1's complement checksum over the byte
1122      * stream. The option to start from an intermediate checksum accumulated
1123      * over another previous byte stream is not implemented. The implementation
1124      * accumulates in two 64-bit integer values the two low-order and the two
1125      * high-order bytes of adjacent 4-byte groups. A carry-over of bits is never
1126      * done within the main loop (only once at the end at reduction to a 32-bit
1127      * positive integer) since an overflow of a 64-bit value (signed, with
1128      * maximum at 2^63-1) by summation of 16-bit values could only occur after
1129      * adding approximately 140G short values (=2^47) (280GBytes) or more. We
1130      * assume for now that this routine here is never called to swallow FITS
1131      * files of that size or larger.
1132      * 
1133      * @param data
1134      *            the byte sequence
1135      * @return the 32bit checksum in the range from 0 to 2^32-1
1136      * @see "http://heasarc.gsfc.nasa.gov/docs/heasarc/fits/checksum.html"
1137      * @author R J Mathar
1138      * @since 2005-10-05
1139      */
1140     public static long checksum(final byte[] data) {
1141         long hi = 0;
1142         long lo = 0;
1143         final int len = 2 * (data.length / 4);
1144         // System.out.println(data.length + " bytes") ;
1145         final int remain = data.length % 4;
1146         /*
1147          * a write(2) on Sparc/PA-RISC would write the MSB first, on Linux the
1148          * LSB; by some kind of coincidence, we can stay with the byte order
1149          * known from the original C version of the algorithm.
1150          */
1151         for (int i = 0; i < len; i += 2) {
1152             /*
1153              * The four bytes in this block handled by a single 'i' are each
1154              * signed (-128 to 127) in Java and need to be masked indivdually to
1155              * avoid sign extension /propagation.
1156              */
1157             hi += (data[2 * i] << 8) & 0xff00L | data[2 * i + 1] & 0xffL;
1158             lo += (data[2 * i + 2] << 8) & 0xff00L | data[2 * i + 3] & 0xffL;
1159         }
1160 
1161         /*
1162          * The following three cases actually cannot happen since FITS records
1163          * are multiples of 2880 bytes.
1164          */
1165         if (remain >= 1) {
1166             hi += (data[2 * len] << 8) & 0xff00L;
1167         }
1168         if (remain >= 2) {
1169             hi += data[2 * len + 1] & 0xffL;
1170         }
1171         if (remain >= 3) {
1172             lo += (data[2 * len + 2] << 8) & 0xff00L;
1173         }
1174 
1175         long hicarry = hi >>> 16;
1176         long locarry = lo >>> 16;
1177         while (hicarry != 0 || locarry != 0) {
1178             hi = (hi & 0xffffL) + locarry;
1179             lo = (lo & 0xffffL) + hicarry;
1180             hicarry = hi >>> 16;
1181             locarry = lo >>> 16;
1182         }
1183         return (hi << 16) + lo;
1184     }

1186     /**
1187      * Encode a 32bit integer according to the Seaman-Pence proposal.
1188      * 
1189      * @param c
1190      *            the checksum previously calculated
1191      * @return the encoded string of 16 bytes.
1192      * @see http
1193      *      ://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/general/checksum/
1194      *      node14.html#SECTION00035000000000000000
1195      * @author R J Mathar
1196      * @since 2005-10-05
1197      */
1198     private static String checksumEnc(final long c, final boolean compl) {
1199         byte[] asc = new byte[16];
1200         final int[] exclude = {
1201             0x3a,
1202             0x3b,
1203             0x3c,
1204             0x3d,
1205             0x3e,
1206             0x3f,
1207             0x40,
1208             0x5b,
1209             0x5c,
1210             0x5d,
1211             0x5e,
1212             0x5f,
1213             0x60
1214         };
1215         final long[] mask = {
1216             0xff000000L,
1217             0xff0000L,
1218             0xff00L,
1219             0xffL
1220         };
1221         final int offset = 0x30; /* ASCII 0 (zero */
1222         final long value = compl ? ~c : c;
1223         for (int i = 0; i < 4; i++) {
1224             final int byt = (int) ((value & mask[i]) >>> (24 - 8 * i)); // each
1225                                                                         // byte
1226                                                                         // becomes
1227                                                                         // four
1228             final int quotient = byt / 4 + offset;
1229             final int remainder = byt % 4;
1230             int[] ch = new int[4];
1231             for (int j = 0; j < 4; j++) {
1232                 ch[j] = quotient;
1233             }
1234 
1235             ch[0] += remainder;
1236             boolean check = true;
1237             for (; check;) // avoid ASCII punctuation
1238             {
1239                 check = false;
1240                 for (int k = 0; k < exclude.length; k++) {
1241                     for (int j = 0; j < 4; j += 2) {
1242                         if (ch[j] == exclude[k] || ch[j + 1] == exclude[k]) {
1243                             ch[j]++;
1244                             ch[j + 1]--;
1245                             check = true;
1246                         }
1247                     }
1248                 }
1249             }
1250 
1251             for (int j = 0; j < 4; j++) // assign the bytes
1252             {
1253                 asc[4 * j + i] = (byte) (ch[j]);
1254             }
1255         }
1256 
1257         // shift the bytes 1 to the right circularly.
1258         try {
1259             String resul = AsciiFuncs.asciiString(asc, 15, 1);
1260             return resul.concat(AsciiFuncs.asciiString(asc, 0, 15));
1261         } catch (Exception e) {
1262             // Impossible I hope
1263             System.err.println("CheckSum Error finding ASCII encoding");
1264             return null;
1265         }
1266     }
