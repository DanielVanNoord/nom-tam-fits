66      /**
67       * Convert a FITS date string to a Java <CODE>Date</CODE> object.
68       * 
69       * @param dStr
70       *            the FITS date
71       * @exception FitsException
72       *                if <CODE>dStr</CODE> does not contain a valid FITS date.
73       */
74      public FitsDate(String dStr) throws FitsException {
75          // if the date string is null, we are done
76          if (dStr == null) {
77              return;
78          }
79  
80          // if the date string is empty, we are done
81          dStr = dStr.trim();
82          if (dStr.length() == 0) {
83              return;
84          }
85  
86          // if string contains at least 8 characters...
87          int len = dStr.length();
88          if (len >= 8) {
89              int first;
90  
91              // ... and there is a "/" in the string...
92              first = dStr.indexOf('-');
93              if (first == 4 && first < len) {
94  
95                  // ... this must be an new-style date
96                  buildNewDate(dStr, first, len);
97  
98                  // no "/" found; maybe it is an old-style date...
99              } else {
100 
101                 first = dStr.indexOf('/');
102                 if (first > 1 && first < len) {
103 
104                     // ... this must be an old-style date
105                     buildOldDate(dStr, first, len);
106                 }
107             }
108         }
109 
110         if (year == -1) {
111             throw new FitsException("Bad FITS date string \"" + dStr + '"');
112         }
113     }
114 
115     private void buildOldDate(String dStr, int first, int len) {
116         int middle = dStr.indexOf('/', first + 1);
117         if (middle > first + 2 && middle < len) {
118 
119             try {
120 
121                 year = Integer.parseInt(dStr.substring(middle + 1)) + 1900;
122                 month = Integer.parseInt(dStr.substring(first + 1, middle));
123                 mday = Integer.parseInt(dStr.substring(0, first));
124 
125             } catch (NumberFormatException e) {
126 
127                 year = month = mday = -1;
128             }
129         }
130     }
131 
132     private void parseTime(String tStr) throws FitsException {
133         int first = tStr.indexOf(':');
134         if (first < 0) {
135             throw new FitsException("Bad time");
136         }
137 
138         int len = tStr.length();
139 
140         int middle = tStr.indexOf(':', first + 1);
141         if (middle > first + 2 && middle < len) {
142 
143             if (middle + 3 < len && tStr.charAt(middle + 3) == '.') {
144                 double d = Double.valueOf(tStr.substring(middle + 3)).doubleValue();
145                 millisecond = (int) (d * 1000);
146 
147                 len = middle + 3;
148             }
149 
150             try {
151                 hour = Integer.parseInt(tStr.substring(0, first));
152                 minute = Integer.parseInt(tStr.substring(first + 1, middle));
153                 second = Integer.parseInt(tStr.substring(middle + 1, len));
154             } catch (NumberFormatException e) {
155                 hour = minute = second = millisecond = -1;
156             }
157         }
158     }
159 
160     private void buildNewDate(String dStr, int first, int len) throws FitsException {
161         // find the middle separator
162         int middle = dStr.indexOf('-', first + 1);
163         if (middle > first + 2 && middle < len) {
164 
165             try {
166 
167                 // if this date string includes a time...
168                 if (middle + 3 < len && dStr.charAt(middle + 3) == 'T') {
169 
170                     // ... try to parse the time
171                     try {
172                         parseTime(dStr.substring(middle + 4));
173                     } catch (FitsException e) {
174                         throw new FitsException("Bad time in FITS date string \"" + dStr + "\"");
175                     }
176 
177                     // we got the time; mark the end of the date string
178                     len = middle + 3;
179                 }
180 
181                 // parse date string
182                 year = Integer.parseInt(dStr.substring(0, first));
183                 month = Integer.parseInt(dStr.substring(first + 1, middle));
184                 mday = Integer.parseInt(dStr.substring(middle + 1, len));
185 
186             } catch (NumberFormatException e) {
187 
188                 // yikes, something failed; reset everything
189                 year = month = mday = hour = minute = second = millisecond = -1;
190             }
191         }
192     }

194     /**
195      * Get a Java Date object corresponding to this FITS date.
196      * 
197      * @return The Java Date object.
198      */
199     public Date toDate() {
200         if (date == null && year != -1) {
201             TimeZone tz = TimeZone.getTimeZone("GMT");
202             GregorianCalendar cal = new GregorianCalendar(tz);
203 
204             cal.set(Calendar.YEAR, year);
205             cal.set(Calendar.MONTH, month - 1);
206             cal.set(Calendar.DAY_OF_MONTH, mday);
207 
208             if (hour == -1) {
209 
210                 cal.set(Calendar.HOUR_OF_DAY, 0);
211                 cal.set(Calendar.MINUTE, 0);
212                 cal.set(Calendar.SECOND, 0);
213                 cal.set(Calendar.MILLISECOND, 0);
214 
215             } else {
216 
217                 cal.set(Calendar.HOUR_OF_DAY, hour);
218                 cal.set(Calendar.MINUTE, minute);
219                 cal.set(Calendar.SECOND, second);
220                 if (millisecond == -1) {
221                     cal.set(Calendar.MILLISECOND, 0);
222                 } else {
223                     cal.set(Calendar.MILLISECOND, millisecond);
224                 }
225             }
226 
227             date = cal.getTime();
228         }
229 
230         return date;
231     }

248     /**
249      * Create FITS format date string. Note that the date is not rounded.
250      * 
251      * @param epoch
252      *            The epoch to be converted to FITS format.
253      * @param timeOfDay
254      *            Should time of day information be included?
255      */
256     public static String getFitsDateString(Date epoch, boolean timeOfDay) {
257 
258         try {
259             GregorianCalendar cal = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
260 
261             cal.setTime(epoch);
262 
263             StringBuffer fitsDate = new StringBuffer();
264             DecimalFormat df = new DecimalFormat("0000");
265             fitsDate.append(df.format(cal.get(Calendar.YEAR)));
266             fitsDate.append("-");
267             df = new DecimalFormat("00");
268 
269             fitsDate.append(df.format(cal.get(Calendar.MONTH) + 1));
270             fitsDate.append("-");
271             fitsDate.append(df.format(cal.get(Calendar.DAY_OF_MONTH)));
272 
273             if (timeOfDay) {
274                 fitsDate.append("T");
275                 fitsDate.append(df.format(cal.get(Calendar.HOUR_OF_DAY)));
276                 fitsDate.append(":");
277                 fitsDate.append(df.format(cal.get(Calendar.MINUTE)));
278                 fitsDate.append(":");
279                 fitsDate.append(df.format(cal.get(Calendar.SECOND)));
280                 fitsDate.append(".");
281                 df = new DecimalFormat("000");
282                 fitsDate.append(df.format(cal.get(Calendar.MILLISECOND)));
283             }
284 
285             return new String(fitsDate);
286 
287         } catch (Exception e) {
288 
289             return new String("");
290         }
291     }

293     public String toString() {
294         if (year == -1) {
295             return "";
296         }
297 
298         StringBuffer buf = new StringBuffer(23);
299         buf.append(year);
300         buf.append('-');
301         if (month < 10) {
302             buf.append('0');
303         }
304         buf.append(month);
305         buf.append('-');
306         if (mday < 10) {
307             buf.append('0');
308         }
309         buf.append(mday);
310 
311         if (hour != -1) {
312 
313             buf.append('T');
314             if (hour < 10) {
315                 buf.append('0');
316             }
317 
318             buf.append(hour);
319             buf.append(':');
320 
321             if (minute < 10) {
322                 buf.append('0');
323             }
324 
325             buf.append(minute);
326             buf.append(':');
327 
328             if (second < 10) {
329                 buf.append('0');
330             }
331             buf.append(second);
332 
333             if (millisecond != -1) {
334                 buf.append('.');
335 
336                 if (millisecond < 100) {
337                     if (millisecond < 10) {
338                         buf.append("00");
339                     } else {
340                         buf.append('0');
341                     }
342                 }
343                 buf.append(millisecond);
344             }
345         }
346 
347         return buf.toString();
348     }

350     public static void testArgs(String args[]) {
351         for (int i = 0; i < args.length; i++) {
352 
353             try {
354                 FitsDate fd = new FitsDate(args[i]);
355                 System.out.println("\"" + args[i] + "\" => " + fd + " => " + fd.toDate());
356             } catch (Exception e) {
357                 System.err.println("Date \"" + args[i] + "\" threw " + e.getClass().getName() + "(" + e.getMessage() + ")");
358             }
359         }
360     }
