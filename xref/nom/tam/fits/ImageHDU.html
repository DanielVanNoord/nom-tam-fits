32  import nom.tam.util.ArrayFuncs;
33  import nom.tam.util.BufferedDataInputStream;
34  import nom.tam.image.StandardImageTiler;

36  /** FITS image header/data unit */
37  public class ImageHDU extends BasicHDU {
38  
39      /**
40       * Build an image HDU using the supplied data.
41       * 
42       * @param h
43       *            the header for the image.
44       * @param d
45       *            the data used in the image.
46       * @exception FitsException
47       *                if there was a problem with the data.
48       */
49      public ImageHDU(Header h, Data d) throws FitsException {
50          myData = d;
51          myHeader = h;
52  
53      }
54  
55      /** Indicate that Images can appear at the beginning of a FITS dataset */
56      protected boolean canBePrimary() {
57          return true;
58      }
59  
60      /** Change the Image from/to primary */
61      protected void setPrimaryHDU(boolean status) {
62  
63          try {
64              super.setPrimaryHDU(status);
65          } catch (FitsException e) {
66              System.err.println("Impossible exception in ImageData");
67          }
68  
69          if (status) {
70              myHeader.setSimple(true);
71          } else {
72              myHeader.setXtension("IMAGE");
73          }
74      }
75  
76      /**
77       * Check that this HDU has a valid header for this type.
78       * 
79       * @return <CODE>true</CODE> if this HDU has a valid header.
80       */
81      public static boolean isHeader(Header hdr) {
82          boolean found = false;
83          found = hdr.getBooleanValue("SIMPLE");
84          if (!found) {
85              String s = hdr.getStringValue("XTENSION");
86              if (s != null) {
87                  if (s.trim().equals("IMAGE") || s.trim().equals("IUEIMAGE")) {
88                      found = true;
89                  }
90              }
91          }
92          if (!found) {
93              return false;
94          }
95          return !hdr.getBooleanValue("GROUPS");
96      }
97  
98      /**
99       * Check if this object can be described as a FITS image.
100      * 
101      * @param o
102      *            The Object being tested.
103      */
104     public static boolean isData(Object o) {
105         String s = o.getClass().getName();
106 
107         int i;
108         for (i = 0; i < s.length(); i += 1) {
109             if (s.charAt(i) != '[') {
110                 break;
111             }
112         }
113 
114         // Allow all non-boolean/Object arrays.
115         // This does not check the rectangularity of the array though.
116         if (i <= 0 || s.charAt(i) == 'L' || s.charAt(i) == 'Z') {
117             return false;
118         } else {
119             return true;
120         }
121     }
122 
123     /**
124      * Create a Data object to correspond to the header description.
125      * 
126      * @return An unfilled Data object which can be used to read in the data for
127      *         this HDU.
128      * @exception FitsException
129      *                if the image extension could not be created.
130      */
131     public Data manufactureData() throws FitsException {
132         return manufactureData(myHeader);
133     }
134 
135     public static Data manufactureData(Header hdr) throws FitsException {
136         return new ImageData(hdr);
137     }
138 
139     /**
140      * Create a header that describes the given image data.
141      * 
142      * @param d
143      *            The image to be described.
144      * @exception FitsException
145      *                if the object does not contain valid image data.
146      */
147     public static Header manufactureHeader(Data d) throws FitsException {
148 
149         if (d == null) {
150             return null;
151         }
152 
153         Header h = new Header();
154         d.fillHeader(h);
155 
156         return h;
157     }
158 
159     /** Encapsulate an object as an ImageHDU. */
160     public static Data encapsulate(Object o) throws FitsException {
161         return new ImageData(o);
162     }
163 
164     public StandardImageTiler getTiler() {
165         return ((ImageData) myData).getTiler();
166     }
167 
168     /**
169      * Print out some information about this HDU.
170      */
171     public void info() {
172         if (isHeader(myHeader)) {
173             System.out.println("  Image");
174         } else {
175             System.out.println("  Image (bad header)");
176         }
177 
178         System.out.println("      Header Information:");
179         System.out.println("         BITPIX=" + myHeader.getIntValue("BITPIX", -1));
180         int naxis = myHeader.getIntValue("NAXIS", -1);
181         System.out.println("         NAXIS=" + naxis);
182         for (int i = 1; i <= naxis; i += 1) {
183             System.out.println("         NAXIS" + i + "=" + myHeader.getIntValue("NAXIS" + i, -1));
184         }
185 
186         System.out.println("      Data information:");
187         try {
188             if (myData.getData() == null) {
189                 System.out.println("        No Data");
190             } else {
191                 System.out.println("         " + ArrayFuncs.arrayDescription(myData.getData()));
192             }
193         } catch (Exception e) {
194             System.out.println("      Unable to get data");
195         }
196     }
197 }
