25  import nom.tam.util.ArrayFuncs;
26  
27  /** Holder for unknown data types. */
28  public class UndefinedHDU extends BasicHDU {
29  
30      /**
31       * Build an image HDU using the supplied data.
32       * 
33       * @param h
34       *            the header for this HDU
35       * @param d
36       *            the data used to build the image.
37       * @exception FitsException
38       *                if there was a problem with the data.
39       */
40      public UndefinedHDU(Header h, Data d) throws FitsException {
41          myData = d;
42          myHeader = h;
43  
44      }
45  
46      /*
47       * Check if we can find the length of the data for this header.
48       * @return <CODE>true</CODE> if this HDU has a valid header.
49       */
50      public static boolean isHeader(Header hdr) {
51          if (hdr.getStringValue("XTENSION") != null && hdr.getIntValue("NAXIS", -1) >= 0) {
52              return true;
53          }
54          return false;
55      }
56  
57      /**
58       * Check if we can use the following object as in an Undefined FITS block.
59       * We allow this so long as computeLSize can get a size. Note that
60       * computeLSize may be wrong!
61       * 
62       * @param o
63       *            The Object being tested.
64       */
65      public static boolean isData(Object o) {
66          return ArrayFuncs.computeLSize(o) > 0;
67      }
68  
69      /**
70       * Create a Data object to correspond to the header description.
71       * 
72       * @return An unfilled Data object which can be used to read in the data for
73       *         this HDU.
74       * @exception FitsException
75       *                if the image extension could not be created.
76       */
77      public Data manufactureData() throws FitsException {
78          return manufactureData(myHeader);
79      }
80  
81      public static Data manufactureData(Header hdr) throws FitsException {
82          return new UndefinedData(hdr);
83      }
84  
85      /**
86       * Create a header that describes the given image data.
87       * 
88       * @param d
89       *            The image to be described.
90       * @exception FitsException
91       *                if the object does not contain valid image data.
92       */
93      public static Header manufactureHeader(Data d) throws FitsException {
94  
95          Header h = new Header();
96          d.fillHeader(h);
97  
98          return h;
99      }
100 
101     /** Encapsulate an object as an ImageHDU. */
102     public static Data encapsulate(Object o) throws FitsException {
103         return new UndefinedData(o);
104     }
105 
106     /**
107      * Print out some information about this HDU.
108      */
109     public void info() {
110 
111         System.out.println("  Unhandled/Undefined/Unknown Type");
112         System.out.println("  XTENSION=" + myHeader.getStringValue("XTENSION").trim());
113         System.out.println("  Apparent size:" + myData.getTrueSize());
114     }
115 }
