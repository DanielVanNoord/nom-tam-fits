25  /**
26   * This class contains the code which associates particular FITS types with
27   * header and data configurations. It comprises a set of Factory methods which
28   * call appropriate methods in the HDU classes. If -- God forbid -- a new FITS
29   * HDU type were created, then the XXHDU, XXData classes would need to be added
30   * and this file modified but no other changes should be needed in the FITS
31   * libraries.
32   */
33  public class FitsFactory {
34  
35      private static boolean useAsciiTables = true;
36  
37      private static boolean useHierarch = false;
38  
39      private static boolean checkAsciiStrings = false;
40  
41      private static boolean allowTerminalJunk = false;
42  
43      /**
44       * Indicate whether ASCII tables should be used where feasible.
45       */
46      public static void setUseAsciiTables(boolean flag) {
47          useAsciiTables = flag;
48      }
49  
50      /** Get the current status of ASCII table writing */
51      static boolean getUseAsciiTables() {
52          return useAsciiTables;
53      }
54  
55      /** Enable/Disable hierarchical keyword processing. */
56      public static void setUseHierarch(boolean flag) {
57          useHierarch = flag;
58      }
59  
60      /**
61       * Enable/Disable checking of strings values used in tables to ensure that
62       * they are within the range specified by the FITS standard. The standard
63       * only allows the values 0x20 - 0x7E with null bytes allowed in one limited
64       * context. Disabled by default.
65       */
66      public static void setCheckAsciiStrings(boolean flag) {
67          checkAsciiStrings = flag;
68      }
69  
70      /** Get the current status for string checking. */
71      static boolean getCheckAsciiStrings() {
72          return checkAsciiStrings;
73      }
74  
75      /** Are we processing HIERARCH style keywords */
76      public static boolean getUseHierarch() {
77          return useHierarch;
78      }
79  
80      /** Do we allow junk after a valid FITS file? */
81      public static void setAllowTerminalJunk(boolean flag) {
82          allowTerminalJunk = flag;
83      }
84  
85      /**
86       * Is terminal junk (i.e., non-FITS data following a valid HDU) allowed.
87       */
88      public static boolean getAllowTerminalJunk() {
89          return allowTerminalJunk;
90      }
91  
92      /**
93       * Given a Header return an appropriate datum.
94       */
95      public static Data dataFactory(Header hdr) throws FitsException {
96  
97          if (ImageHDU.isHeader(hdr)) {
98              Data d = ImageHDU.manufactureData(hdr);
99              hdr.afterExtend(); // Fix for positioning error noted by V. Forchi
100             return d;
101         } else if (RandomGroupsHDU.isHeader(hdr)) {
102             return RandomGroupsHDU.manufactureData(hdr);
103         } else if (useAsciiTables && AsciiTableHDU.isHeader(hdr)) {
104             return AsciiTableHDU.manufactureData(hdr);
105         } else if (BinaryTableHDU.isHeader(hdr)) {
106             return BinaryTableHDU.manufactureData(hdr);
107         } else if (UndefinedHDU.isHeader(hdr)) {
108             return UndefinedHDU.manufactureData(hdr);
109         } else {
110             throw new FitsException("Unrecognizable header in dataFactory");
111         }
112 
113     }
114 
115     /**
116      * Given an object, create the appropriate FITS header to describe it.
117      * 
118      * @param o
119      *            The object to be described.
120      */
121     public static BasicHDU HDUFactory(Object o) throws FitsException {
122         Data d;
123         Header h;
124 
125         if (o instanceof Header) {
126             h = (Header) o;
127             d = dataFactory(h);
128 
129         } else if (ImageHDU.isData(o)) {
130             d = ImageHDU.encapsulate(o);
131             h = ImageHDU.manufactureHeader(d);
132         } else if (RandomGroupsHDU.isData(o)) {
133             d = RandomGroupsHDU.encapsulate(o);
134             h = RandomGroupsHDU.manufactureHeader(d);
135         } else if (useAsciiTables && AsciiTableHDU.isData(o)) {
136             d = AsciiTableHDU.encapsulate(o);
137             h = AsciiTableHDU.manufactureHeader(d);
138         } else if (BinaryTableHDU.isData(o)) {
139             d = BinaryTableHDU.encapsulate(o);
140             h = BinaryTableHDU.manufactureHeader(d);
141         } else if (UndefinedHDU.isData(o)) {
142             d = UndefinedHDU.encapsulate(o);
143             h = UndefinedHDU.manufactureHeader(d);
144         } else {
145             throw new FitsException("Invalid data presented to HDUFactory");
146         }
147 
148         return HDUFactory(h, d);
149 
150     }
151 
152     /**
153      * Given Header and data objects return the appropriate type of HDU.
154      */
155     public static BasicHDU HDUFactory(Header hdr, Data d) throws FitsException {
156 
157         if (d instanceof ImageData) {
158             return new ImageHDU(hdr, d);
159         } else if (d instanceof RandomGroupsData) {
160             return new RandomGroupsHDU(hdr, d);
161         } else if (d instanceof AsciiTable) {
162             return new AsciiTableHDU(hdr, d);
163         } else if (d instanceof BinaryTable) {
164             return new BinaryTableHDU(hdr, d);
165         } else if (d instanceof UndefinedData) {
166             return new UndefinedHDU(hdr, d);
167         }
168 
169         return null;
170     }
