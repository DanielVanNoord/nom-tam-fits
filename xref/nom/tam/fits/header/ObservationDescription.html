32  /**
33   * File checksum keywords. This data dictionary contains FITS keywords that have
34   * been widely used within the astronomical community. It is recommended that
35   * these keywords only be used as defined here. These are the Keywords that
36   * describe the observation. {@link http
37   * ://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html}
38   * 
39   * @author Richard van Nieuwenhoven
40   */
41  public enum ObservationDescription implements IFitsHeader {
42      /**
43       * The value field shall contain a floating point number giving the angle
44       * between the direction of the observation (e.g., the optical axis of the
45       * telescope or the position of the target) and the sun, measured in
46       * degrees.
47       */
48      SUNANGLE(STATUS.STScI, HDU.ANY, VALUE.REAL, "angle between the observation and the sun"),
49      /**
50       * The value field shall contain a floating point number giving the angle
51       * between the direction of the observation (e.g., the optical axis of the
52       * telescope or the position of the target) and the moon, measured in
53       * degrees.
54       */
55      MOONANGL(STATUS.STScI, HDU.ANY, VALUE.REAL, "angle between the observation and the moon"),
56      /**
57       * The value field gives the Right Ascension of the observation. It may be
58       * expressed either as a floating point number in units of decimal degrees,
59       * or as a character string in 'HH:MM:SS.sss' format where the decimal point
60       * and number of fractional digits are optional. The coordinate reference
61       * frame is given by the RADECSYS keyword, and the coordinate epoch is given
62       * by the EQUINOX keyword. Example: 180.6904 or '12:02:45.7'.
63       */
64      RA(STATUS.NOAO, HDU.ANY, VALUE.STRING, "R.A. of the observation"),
65      /**
66       * The value field gives the declination of the observation. It may be
67       * expressed either as a floating point number in units of decimal degrees,
68       * or as a character string in 'dd:mm:ss.sss' format where the decimal point
69       * and number of fractional digits are optional. The coordinate reference
70       * frame is given by the RADECSYS keyword, and the coordinate epoch is given
71       * by the EQUINOX keyword. Example: -47.25944 or '-47:15:34.00'.
72       */
73      DEC(STATUS.NOAO, HDU.ANY, VALUE.STRING, "declination of the observed object"),
74      RA_NOM(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
75      DEC_NOM(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
76      RA_OBJ(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
77      DEC_OBJ(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
78      RA_PNT(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
79      DEC_PNT(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
80      PA_PNT(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
81      RA_SCX(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
82      DEC_SCX(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
83      RA_SCY(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
84      DEC_SXY(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
85      RA_SCZ(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
86      DEC_SCZ(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
87      ORIENTAT(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
88      AIRMASS(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
89      LATITUDE(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
90      OBJNAME(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, ""),
91      OBS_ID(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, "");
92  
93      private IFitsHeader key;
94  
95      private ObservationDescription(IFitsHeader.STATUS status, HDU hdu, VALUE valueType, String comment) {
96          this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
97      }
98  
99      @Override
100     public String key() {
101         return key.key();
102     }
103 
104     @Override
105     public STATUS status() {
106         return key.status();
107     }
108 
109     @Override
110     public HDU hdu() {
111         return key.hdu();
112     }
113 
114     @Override
115     public VALUE valueType() {
116         return key.valueType();
117     }
118 
119     @Override
120     public String comment() {
121         return key.comment();
122     }
123 
124     @Override
125     public IFitsHeader n(int number) {
126         return key.n(number);
127     }
128 }
