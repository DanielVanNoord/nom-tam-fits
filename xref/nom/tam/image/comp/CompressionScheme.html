36  import java.io.IOException;
37  import java.io.InputStream;
38  import java.io.OutputStream;
39  import java.util.Map;
40  import nom.tam.fits.FitsException;
41  import nom.tam.fits.Header;
42  
43  /**
44   * @author tmcglynn
45   */
46  public interface CompressionScheme {
47  
48      /** Return the 'name' of the compression scheme */
49      public abstract String name();
50  
51      /**
52       * Initialize the compression scheme with any appropriate parameters.
53       */
54      public abstract void initialize(Map<String, String> params);
55  
56      /** Return a stream which compresses the input. */
57      public abstract byte[] compress(byte[] in) throws IOException;
58  
59      /** Return a stream with decompresses the input. */
60      public abstract byte[] decompress(byte[] in, int length) throws IOException;
61  
62      /**
63       * Update the FITS header and compression parameterswith information about
64       * the compression
65       */
66      public abstract void updateForWrite(Header hdr, Map<String, String> parameters) throws FitsException;
67  
68      /** Get the parameters indicated by a given header */
69      public abstract void getParameters(Map<String, String> params, Header hdr);
70  
71  }
