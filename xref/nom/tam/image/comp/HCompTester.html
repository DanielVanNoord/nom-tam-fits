32  import java.io.ByteArrayOutputStream;
33  import java.io.File;
34  import java.io.FileInputStream;
35  import java.io.IOException;
36  import java.util.HashMap;
37  import java.util.Map;
38  import nom.tam.fits.Fits;
39  import nom.tam.fits.FitsFactory;
40  import nom.tam.util.ArrayFuncs;
41  import nom.tam.util.BufferedDataOutputStream;
42  import nom.tam.util.BufferedFile;
43  
44  /**
45   * @author tmcglynn
46   */
47  public class HCompTester {
48  
49      public static void main(String[] args) throws Exception {
50          File f = new File(args[0]);
51          long len = f.length();
52          FileInputStream fs = new FileInputStream(args[0]);
53          byte[] input = new byte[(int) len];
54          int need = (int) len;
55          int have = 0;
56          while (need > 0) {
57              int got = fs.read(input, have, need);
58              if (got > 0) {
59                  need -= got;
60                  have += got;
61              } else {
62                  throw new IOException("Got was:" + got + " with need: " + need);
63              }
64          }
65          HCompress hc = new HCompress();
66          int[] a = hc.decomp(input);
67          System.err.println("Number of ints:" + a.length);
68          int[] dims = new int[]{
69              hc.getHeight(),
70              hc.getWidth()
71          };
72  
73          int[][] img = (int[][]) ArrayFuncs.curl(a, dims);
74          Fits ff = new Fits();
75          ff.addHDU(FitsFactory.HDUFactory(img));
76          BufferedFile bf = new BufferedFile(args[1], "rw");
77          ff.write(bf);
78          bf.close();
79  
80          bf = new BufferedFile("int.out", "rw");
81          bf.write(a);
82          bf.close();
83  
84          ByteArrayOutputStream bd = new ByteArrayOutputStream();
85          BufferedDataOutputStream out = new BufferedDataOutputStream(bd);
86          out.write(a);
87          out.close();
88  
89          byte[] data = bd.toByteArray();
90          System.err.println("Number of bytes:" + data.length);
91  
92          Map<String, String> mp = new HashMap<String, String>();
93          mp.put("nx", "" + img[0].length);
94          mp.put("ny", "" + img.length);
95          mp.put("scale", "" + 900);
96          hc.initialize(mp);
97          byte[] cmp = hc.compress(data);
98          int sum = 0;
99          for (int i = 0; i < cmp.length; i += 1) {
100             sum += cmp[i] & 0xff;
101         }
102         System.out.println("Sum is:" + sum);
103         bf = new BufferedFile(args[2], "rw");
104         bf.write(cmp);
105         bf.close();
106     }
