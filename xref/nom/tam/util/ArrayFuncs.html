47      /**
48       * Compute the size of an object. Note that this only handles arrays or
49       * scalars of the primitive objects and Strings. It returns 0 for any object
50       * array element it does not understand.
51       * 
52       * @param o
53       *            The object whose size is desired.
54       * @deprecated May silently underestimate the size if the size > 2 GB.
55       */
56      public static int computeSize(Object o) {
57          return (int) computeLSize(o);
58      }

60      public static long computeLSize(Object o) {
61  
62          if (o == null) {
63              return 0;
64          }
65  
66          long size = 0;
67          String classname = o.getClass().getName();
68          if (classname.substring(0, 2).equals("[[")) {
69  
70              for (int i = 0; i < ((Object[]) o).length; i += 1) {
71                  size += computeLSize(((Object[]) o)[i]);
72              }
73              return size;
74          }
75  
76          if (classname.charAt(0) == '[' && classname.charAt(1) != 'L') {
77              char c = classname.charAt(1);
78  
79              for (int i = 0; i < PrimitiveInfo.suffixes.length; i += 1) {
80                  if (c == PrimitiveInfo.suffixes[i]) {
81                      return (long) (Array.getLength(o)) * PrimitiveInfo.sizes[i];
82                  }
83              }
84              return 0;
85          }
86  
87          // Do we have a non-primitive array?
88          if (classname.charAt(0) == '[') {
89              int len = 0;
90              for (int i = 0; i < Array.getLength(o); i += 1) {
91                  len += computeLSize(Array.get(o, i));
92              }
93              return len;
94          }
95  
96          // Now a few special scalar objects.
97          if (classname.substring(0, 10).equals("java.lang.")) {
98              classname = classname.substring(10, classname.length());
99              if (classname.equals("Integer") || classname.equals("Float")) {
100                 return 4;
101             } else if (classname.equals("Double") || classname.equals("Long")) {
102                 return 8;
103             } else if (classname.equals("Short") || classname.equals("Char")) {
104                 return 2;
105             } else if (classname.equals("Byte") || classname.equals("Boolean")) {
106                 return 1;
107             } else if (classname.equals("String")) {
108                 return ((String) o).length();
109             } else {
110                 return 0;
111             }
112         }
113         return 0;
114     }

137         if (classname.charAt(1) == '[') {

152     /**
153      * Try to create a deep clone of an Array or a standard clone of a scalar.
154      * The object may comprise arrays of any primitive type or any Object type
155      * which implements Cloneable. However, if the Object is some kind of
156      * collection, e.g., a Vector then only a shallow copy of that object is
157      * made. I.e., deep refers only to arrays.
158      * 
159      * @param o
160      *            The object to be copied.
161      */
162     public static Object deepClone(Object o) {
163 
164         if (o == null) {
165             return null;
166         }
167 
168         String classname = o.getClass().getName();
169 
170         // Is this an array?
171         if (classname.charAt(0) != '[') {
172             return genericClone(o);
173         }
174 
175         // Check if this is a 1D primitive array.
176         if (classname.charAt(1) != '[' && classname.charAt(1) != 'L') {
177             try {
178                 // Some compilers (SuperCede, e.g.) still
179                 // think you have to catch this...
180                 if (false) {
181                     throw new CloneNotSupportedException();
182                 }
183                 switch (classname.charAt(1)) {
184                     case 'B':
185                         return ((byte[]) o).clone();
186                     case 'Z':
187                         return ((boolean[]) o).clone();
188                     case 'C':
189                         return ((char[]) o).clone();
190                     case 'S':
191                         return ((short[]) o).clone();
192                     case 'I':
193                         return ((int[]) o).clone();
194                     case 'J':
195                         return ((long[]) o).clone();
196                     case 'F':
197                         return ((float[]) o).clone();
198                     case 'D':
199                         return ((double[]) o).clone();
200                     default:
201                         System.err.println("Unknown primtive array class:" + classname);
202                         return null;
203 
204                 }
205             } catch (CloneNotSupportedException e) {
206             }
207         }
208 
209         // Get the base type.
210         int ndim = 1;
211         while (classname.charAt(ndim) == '[') {
212             ndim += 1;
213         }
214         Class baseClass;
215         if (classname.charAt(ndim) != 'L') {
216             baseClass = getBaseClass(o);
217         } else {
218             try {
219                 baseClass = Class.forName(classname.substring(ndim + 1, classname.length() - 1));
220             } catch (ClassNotFoundException e) {
221                 System.err.println("Internal error: class definition inconsistency: " + classname);
222                 return null;
223             }
224         }
225 
226         // Allocate the array but make all but the first dimension 0.
227         int[] dims = new int[ndim];
228         dims[0] = Array.getLength(o);
229         for (int i = 1; i < ndim; i += 1) {
230             dims[i] = 0;
231         }
232 
233         Object copy = ArrayFuncs.newInstance(baseClass, dims);
234 
235         // Now fill in the next level down by recursion.
236         for (int i = 0; i < dims[0]; i += 1) {
237             Array.set(copy, i, deepClone(Array.get(o, i)));
238         }
239 
240         return copy;
241     }

243     /**
244      * Clone an Object if possible. This method returns an Object which is a
245      * clone of the input object. It checks if the method implements the
246      * Cloneable interface and then uses reflection to invoke the clone method.
247      * This can't be done directly since as far as the compiler is concerned the
248      * clone method for Object is protected and someone could implement
249      * Cloneable but leave the clone method protected. The cloning can fail in a
250      * variety of ways which are trapped so that it returns null instead. This
251      * method will generally create a shallow clone. If you wish a deep copy of
252      * an array the method deepClone should be used.
253      * 
254      * @param o
255      *            The object to be cloned.
256      */
257     public static Object genericClone(Object o) {
258 
259         if (!(o instanceof Cloneable)) {
260             return null;
261         }
262 
263         Class[] argTypes = new Class[0];
264         Object[] args = new Object[0];
265         Class type = o.getClass();
266 
267         try {
268             return type.getMethod("clone", argTypes).invoke(o, args);
269         } catch (Exception e) {
270             if (type.isArray()) {
271                 return deepClone(o);
272             }
273             // Implements cloneable, but does not
274             // apparently make clone public.
275             return null;
276         }
277     }

279     /**
280      * Copy one array into another. This function copies the contents of one
281      * array into a previously allocated array. The arrays must agree in type
282      * and size.
283      * 
284      * @param original
285      *            The array to be copied.
286      * @param copy
287      *            The array to be copied into. This array must already be fully
288      *            allocated.
289      */
290     public static void copyArray(Object original, Object copy) {
291         String oname = original.getClass().getName();
292         String cname = copy.getClass().getName();
293 
294         if (!oname.equals(cname)) {
295             return;
296         }
297 
298         if (oname.charAt(0) != '[') {
299             return;
300         }
301 
302         if (oname.charAt(1) == '[') {
303             Object[] x = (Object[]) original;
304             Object[] y = (Object[]) copy;
305             if (x.length != y.length) {
306                 return;
307             }
308             for (int i = 0; i < x.length; i += 1) {
309                 copyArray(x, y);
310             }
311         }
312         int len = Array.getLength(original);
313 
314         System.arraycopy(original, 0, copy, 0, len);
315     }

317     /**
318      * Find the dimensions of an object. This method returns an integer array
319      * with the dimensions of the object o which should usually be an array. It
320      * returns an array of dimension 0 for scalar objects and it returns -1 for
321      * dimension which have not been allocated, e.g., int[][][] x = new
322      * int[100][][]; should return [100,-1,-1].
323      * 
324      * @param o
325      *            The object to get the dimensions of.
326      */
327     public static int[] getDimensions(Object o) {
328 
329         if (o == null) {
330             return null;
331         }
332 
333         String classname = o.getClass().getName();
334 
335         int ndim = 0;
336 
337         while (classname.charAt(ndim) == '[') {
338             ndim += 1;
339         }
340 
341         int[] dimens = new int[ndim];
342 
343         for (int i = 0; i < ndim; i += 1) {
344             dimens[i] = -1; // So that we can distinguish a null from a 0
345                             // length.
346         }
347 
348         for (int i = 0; i < ndim; i += 1) {
349             dimens[i] = java.lang.reflect.Array.getLength(o);
350             if (dimens[i] == 0) {
351                 return dimens;
352             }
353             if (i != ndim - 1) {
354                 o = ((Object[]) o)[0];
355                 if (o == null) {
356                     return dimens;
357                 }
358             }
359         }
360         return dimens;
361     }

377     /**
378      * This routine returns the base class of an object. This is just the class
379      * of the object for non-arrays.
380      */
381     public static Class getBaseClass(Object o) {
382 
383         if (o == null) {
384             return Void.TYPE;
385         }
386 
387         String className = o.getClass().getName();
388 
389         int dims = 0;
390         while (className.charAt(dims) == '[') {
391             dims += 1;
392         }
393 
394         if (dims == 0) {
395             return o.getClass();
396         }
397 
398         char c = className.charAt(dims);
399         for (int i = 0; i < PrimitiveInfo.suffixes.length; i += 1) {
400             if (c == PrimitiveInfo.suffixes[i]) {
401                 return PrimitiveInfo.classes[i];
402             }
403         }
404 
405         if (c == 'L') {
406             try {
407                 return Class.forName(className.substring(dims + 1, className.length() - 1));
408             } catch (ClassNotFoundException e) {
409                 return null;
410             }
411         }
412         return null;
413     }

470     /**
471      * Just create a simple pattern cycling through valid byte values. We use
472      * bytes because they can be cast to any other numeric type.
473      * 
474      * @param o
475      *            The array in which the test pattern is to be set.
476      * @param start
477      *            The value for the first element.
478      */
479     public static byte testPattern(Object o, byte start) {
480 
481         int[] dims = getDimensions(o);
482         if (dims.length > 1) {
483             for (int i = 0; i < ((Object[]) o).length; i += 1) {
484                 start = testPattern(((Object[]) o)[i], start);
485             }
486 
487         } else if (dims.length == 1) {
488             for (int i = 0; i < dims[0]; i += 1) {
489                 java.lang.reflect.Array.setByte(o, i, start);
490                 start += 1;
491             }
492         }
493         return start;
494     }

496     /**
497      * Generate a description of an array (presumed rectangular).
498      * 
499      * @param o
500      *            The array to be described.
501      */
502     public static String arrayDescription(Object o) {
503 
504         Class base = getBaseClass(o);
505         if (base == Void.TYPE) {
506             return "NULL";
507         }
508 
509         int[] dims = getDimensions(o);
510 
511         StringBuffer desc = new StringBuffer();
512 
513         // Note that all instances Class describing a given class are
514         // the same so we can use == here.
515         boolean found = false;
516 
517         for (int i = 0; i < PrimitiveInfo.classes.length; i += 1) {
518             if (base == PrimitiveInfo.classes[i]) {
519                 found = true;
520                 desc.append(PrimitiveInfo.types[i]);
521                 break;
522             }
523         }
524 
525         if (!found) {
526             desc.append(base.getName());
527         }
528 
529         if (dims != null) {
530             desc.append("[");
531             for (int i = 0; i < dims.length; i += 1) {
532                 desc.append("" + dims[i]);
533                 if (i < dims.length - 1) {
534                     desc.append("][");
535                 }
536             }
537             desc.append("]");
538         }
539         return new String(desc);
540     }

544      * 
545      * @param o
546      *            The array to be examined.
547      */
548     public static void examinePrimitiveArray(Object o) {
549         String className = o.getClass().getName();
550 
551         // If we have a two-d array, or if the array is a one-d array
552         // of Objects, then recurse over the next dimension. We handle

555             System.out.println("[");

557                 examinePrimitiveArray(((Object[]) o)[i]);
558             }
559             System.out.print("]");
560         } else if (className.charAt(0) != '[') {
561             System.out.println(className);
562         } else {
563             System.out.println("[" + java.lang.reflect.Array.getLength(o) + "]" + className.substring(1));
564         }
565     }

567     /**
568      * Given an array of arbitrary dimensionality return the array flattened
569      * into a single dimension.
570      * 
571      * @param input
572      *            The input array.
573      */
574     public static Object flatten(Object input) {
575 
576         int[] dimens = getDimensions(input);
577         if (dimens.length <= 1) {
578             return input;
579         }
580         int size = 1;
581         for (int i = 0; i < dimens.length; i += 1) {
582             size *= dimens[i];
583         }
584 
585         Object flat = ArrayFuncs.newInstance(getBaseClass(input), size);
586 
587         if (size == 0) {
588             return flat;
589         }
590 
591         int offset = 0;
592         doFlatten(input, flat, offset);
593         return flat;
594     }

596     /**
597      * This routine does the actually flattening of multi-dimensional arrays.
598      * 
599      * @param input
600      *            The input array to be flattened.
601      * @param output
602      *            The flattened array.
603      * @param offset
604      *            The current offset within the output array.
605      * @return The number of elements within the array.
606      */
607     protected static int doFlatten(Object input, Object output, int offset) {
608 
609         String classname = input.getClass().getName();
610         if (classname.charAt(0) != '[') {
611             throw new RuntimeException("Attempt to flatten non-array");
612         }
613         int size = Array.getLength(input);
614 
615         if (classname.charAt(1) != '[') {
616             System.arraycopy(input, 0, output, offset, size);
617             return size;
618         }
619         int total = 0;
620         Object[] xx = (Object[]) input;
621         for (int i = 0; i < size; i += 1) {
622             int len = doFlatten(xx[i], output, offset + total);
623             total += len;
624         }
625         return total;
626     }

628     /**
629      * Curl an input array up into a multi-dimensional array.
630      * 
631      * @param input
632      *            The one dimensional array to be curled.
633      * @param dimens
634      *            The desired dimensions
635      * @return The curled array.
636      */
637     public static Object curl(Object input, int[] dimens) {
638 
639         if (input == null) {
640             return null;
641         }
642         String classname = input.getClass().getName();
643         if (classname.charAt(0) != '[' || classname.charAt(1) == '[') {
644             throw new RuntimeException("Attempt to curl non-1D array");
645         }
646 
647         int size = Array.getLength(input);
648 
649         int test = 1;
650         for (int i = 0; i < dimens.length; i += 1) {
651             test *= dimens[i];
652         }
653 
654         if (test != size) {
655             throw new RuntimeException("Curled array does not fit desired dimensions");
656         }
657 
658         Class base = getBaseClass(input);
659 
660         Object newArray = ArrayFuncs.newInstance(base, dimens);
661 
662         int offset = 0;
663 
664         doCurl(input, newArray, dimens, offset);
665         return newArray;
666 
667     }

669     /**
670      * Do the curling of the 1-d to multi-d array.
671      * 
672      * @param input
673      *            The 1-d array to be curled.
674      * @param output
675      *            The multi-dimensional array to be filled.
676      * @param dimens
677      *            The desired output dimensions.
678      * @param offset
679      *            The current offset in the input array.
680      * @return The number of elements curled.
681      */
682     protected static int doCurl(Object input, Object output, int[] dimens, int offset) {
683 
684         if (dimens.length == 1) {
685             System.arraycopy(input, offset, output, 0, dimens[0]);
686             return dimens[0];
687         }
688 
689         int total = 0;
690         int[] xdimens = new int[dimens.length - 1];
691         for (int i = 1; i < dimens.length; i += 1) {
692             xdimens[i - 1] = dimens[i];
693         }
694 
695         for (int i = 0; i < dimens[0]; i += 1) {
696             total += doCurl(input, ((Object[]) output)[i], xdimens, offset + total);
697         }
698         return total;
699     }

701     /**
702      * Create an array of a type given by new type with the dimensionality given
703      * in array.
704      * 
705      * @param array
706      *            A possibly multidimensional array to be converted.
707      * @param newType
708      *            The desired output type. This should be one of the class
709      *            descriptors for primitive numeric data, e.g., double.type.
710      */
711     public static Object mimicArray(Object array, Class newType) {
712 
713         String classname = array.getClass().getName();
714         if (classname.charAt(0) != '[') {
715             return null;
716         }
717 
718         int dims = 1;
719 
720         while (classname.charAt(dims) == '[') {
721             dims += 1;
722         }
723 
724         Object mimic;
725 
726         if (dims > 1) {
727 
728             Object[] xarray = (Object[]) array;
729             int[] dimens = new int[dims];
730             dimens[0] = xarray.length; // Leave other dimensions at 0.
731 
732             mimic = ArrayFuncs.newInstance(newType, dimens);
733 
734             for (int i = 0; i < xarray.length; i += 1) {
735                 Object temp = mimicArray(xarray[i], newType);
736                 ((Object[]) mimic)[i] = temp;
737             }
738 
739         } else {
740             mimic = ArrayFuncs.newInstance(newType, Array.getLength(array));
741         }
742 
743         return mimic;
744     }

1211     /**
1212      * Are two objects equal? Arrays have the standard object equals method
1213      * which only returns true if the two object are the same. This method
1214      * returns true if every element of the arrays match. The inputs may be of
1215      * any dimensionality. The dimensionality and dimensions of the arrays must
1216      * match as well as any elements. If the elements are non-primitive.
1217      * non-array objects, then the equals method is called for each element. If
1218      * both elements are multi-dimensional arrays, then the method recurses.
1219      */
1220     public static boolean arrayEquals(Object x, Object y, double tolf, double told) {
1221 
1222         // Handle the special cases first.
1223         // We treat null == null so that two object arrays
1224         // can match if they have matching null elements.
1225         if (x == null && y == null) {
1226             return true;
1227         }
1228 
1229         if (x == null || y == null) {
1230             return false;
1231         }
1232 
1233         Class xClass = x.getClass();
1234         Class yClass = y.getClass();
1235 
1236         if (xClass != yClass) {
1237             return false;
1238         }
1239 
1240         if (!xClass.isArray()) {
1241             return x.equals(y);
1242 
1243         } else {
1244             if (xClass.equals(int[].class)) {
1245                 return Arrays.equals((int[]) x, (int[]) y);
1246 
1247             } else if (xClass.equals(double[].class)) {
1248                 if (told == 0) {
1249                     return Arrays.equals((double[]) x, (double[]) y);
1250                 } else {
1251                     return doubleArrayEquals((double[]) x, (double[]) y, told);
1252                 }
1253 
1254             } else if (xClass.equals(long[].class)) {
1255                 return Arrays.equals((long[]) x, (long[]) y);
1256 
1257             } else if (xClass.equals(float[].class)) {
1258                 if (tolf == 0) {
1259                     return Arrays.equals((float[]) x, (float[]) y);
1260                 } else {
1261                     return floatArrayEquals((float[]) x, (float[]) y, (float) tolf);
1262                 }
1263 
1264             } else if (xClass.equals(byte[].class)) {
1265                 return Arrays.equals((byte[]) x, (byte[]) y);
1266 
1267             } else if (xClass.equals(short[].class)) {
1268                 return Arrays.equals((short[]) x, (short[]) y);
1269 
1270             } else if (xClass.equals(char[].class)) {
1271                 return Arrays.equals((char[]) x, (char[]) y);
1272 
1273             } else if (xClass.equals(boolean[].class)) {
1274                 return Arrays.equals((boolean[]) x, (boolean[]) y);
1275 
1276             } else {
1277                 // Non-primitive and multidimensional arrays can be
1278                 // cast to Object[]
1279                 Object[] xo = (Object[]) x;
1280                 Object[] yo = (Object[]) y;
1281                 if (xo.length != yo.length) {
1282                     return false;
1283                 }
1284                 for (int i = 0; i < xo.length; i += 1) {
1285                     if (!arrayEquals(xo[i], yo[i], tolf, told)) {
1286                         return false;
1287                     }
1288                 }
1289 
1290                 return true;
1291 
1292             }
1293         }
1294     }
