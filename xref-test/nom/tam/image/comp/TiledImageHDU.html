87          /**
88           * Fill a subset from the tiles
89           */
90          @Override
91          public void getTile(Object array, int[] corners, int[] lengths) throws IOException {
92              System.err.println("Getting tile");
93              // First compute the tiles that we are going to loop over.
94              int[] tFirst = new int[TiledImageHDU.this.naxis];
95              int[] tCount = new int[TiledImageHDU.this.naxis];
96              for (int i = 0; i < TiledImageHDU.this.naxis; i += 1) {
97                  if (corners[i] < 0 || corners[i] >= TiledImageHDU.this.imageSize[i] || lengths[i] <= 0 || corners[i] + lengths[i] > TiledImageHDU.this.imageSize[i]) {
98                      throw new IllegalArgumentException("Invalid tile request");
99                  }
100                 tFirst[i] = corners[i] / TiledImageHDU.this.tileSize[i];
101                 tCount[i] = (corners[i] + lengths[i] - 1) / TiledImageHDU.this.tileSize[i] - tFirst[i] + 1;
102             }
103 
104             // Create a tile looper that goes over the tiles we want.
105             TileLooper tl = new TileLooper(TiledImageHDU.this.imageSize, TiledImageHDU.this.tileSize, tFirst, tCount);
106             String cName = array.getClass().getName();
107             ArrayFuncs.getBaseClass(array);
108 
109             for (TileDescriptor td : tl) {
110                 try {
111                     Object tileData = TiledImageHDU.this.getTile(td);
112                     insertTile(tileData, td.corner, td.size, array, corners, lengths, cName, 0);
113                 } catch (FitsException e) {
114                     throw new IOException("FITS error reading tile", e);
115                 }
116             }
117         }
118     }

251     /**
252      * See if an existing binary table can be treated at a TiledImageHDU.
253      * 
254      * @param input
255      *            A binary table that has been created/read in that may be a
256      *            tiled image.
257      * @throws FitsException
258      *             if the input cannot be treated as a TiledImageHDU.
259      */
260     public TiledImageHDU(BinaryTableHDU input) throws FitsException {
261         super(input.getHeader(), input.getData());
262         this.hdr = input.getHeader();
263         if (!this.hdr.getBooleanValue("ZIMAGE", false) || this.hdr.getStringValue("ZCMPTYPE") == null || this.hdr.getIntValue("ZBITPIX", -1) == -1
264                 || this.hdr.getIntValue("ZNAXIS", -1) == -1) {
265             throw new FitsException("Required keywords not found for TiledImageHDU");
266         }
267         this.naxis = this.hdr.getIntValue("ZNAXIS");
268         this.tileSize = new int[this.naxis];
269         this.imageSize = new int[this.naxis];
270         for (int i = 0; i < this.naxis; i += 1) {
271             String axis = "ZNAXIS" + (i + 1);
272             this.imageSize[i] = this.hdr.getIntValue(axis, -1);
273             if (this.imageSize[i] == -1) {
274                 throw new FitsException("Missing " + axis + " keyword for TileImageHDU");
275             }
276 
277             String tile = "ZTILE" + (i + 1);
278             this.tileSize[i] = this.hdr.getIntValue(tile, -1);
279             // Default tiling is row by row.
280             if (this.tileSize[i] == -1) {
281                 if (i == 0) {
282                     this.tileSize[i] = this.imageSize[i];
283                 } else {
284                     this.tileSize[i] = 1;
285                 }
286             }
287         }
288         this.zbitpix = this.hdr.getIntValue("ZBITPIX");
289         this.baseClass = TiledImageHDU.bitpixClasses.get(this.zbitpix);
290 
291         this.cs = getCompression(this.hdr.getStringValue("ZCMPTYPE"));
292         if (this.hdr.containsKey("ZQUANTIZ")) {
293             if (this.hdr.getStringValue("ZQUANTIZ").toUpperCase().equals("SUBTRACTIVE_DITHER_1")) {
294                 double scale = this.hdr.getDoubleValue("ZSCALE");
295                 double offset = this.hdr.getDoubleValue("ZZERO");
296                 this.quant = new Quantizer(scale, offset);
297             }
298         }
299 
300         Map<String, String> params = getParameters();
301         this.cs.getParameters(params, this.hdr);
302 
303         this.cs.initialize(params);
304 
305     }

307     /**
308      * Create a tiled image HDU from an existing Image HDU.
309      */
310     public TiledImageHDU(ImageHDU input, Map<String, String> parameters) throws FitsException, IOException {
311         super(coreHeader(), nilData());
312 
313         this.hdr = getHeader(); // Get a local reference to the Header.
314         String comp = parameters.get("compression");
315 
316         this.imageSize = input.getAxes();
317         this.naxis = this.imageSize.length;
318         if (this.naxis == 0 || this.imageSize[0] == 0) {
319             throw new FitsException("Cannot compress nil image");
320         }
321         String tiling = parameters.get("tiling");
322         if (tiling == null) {
323             tiling = this.imageSize[0] + "";
324             for (int i = 1; i < this.imageSize.length; i += 1) {
325                 tiling += ",1";
326             }
327         }
328         String[] fields = tiling.split(",");
329         if (fields.length != this.imageSize.length) {
330             throw new FitsException("Tile dimensionality (" + fields.length + ") must match image (" + this.imageSize.length + ")");
331         }
332         this.tileSize = new int[this.imageSize.length];
333         for (int i = 0; i < this.imageSize.length; i += 1) {
334             this.tileSize[i] = Integer.parseInt(fields[i].trim());
335         }
336 
337         Header old = input.getHeader();
338         // Position the insertion pointer after the TFORM1.
339         this.hdr.getStringValue("TFORM1");
340 
341         this.cs = getCompression(comp);
342         insertTileKeywords(old, this.cs, parameters, this.imageSize, this.tileSize);
343 
344         Object kern = input.getKernel();
345 
346         int bitpix = old.getIntValue("BITPIX");
347         this.zbitpix = bitpix;
348         this.baseClass = TiledImageHDU.bitpixClasses.get(this.zbitpix);
349         if (bitpix < 0) {
350             RealStats rs = new RealStats(kern);
351             double offset = rs.min;
352             double scale = rs.noise3 / 16;
353             double bits = Math.log((rs.max - rs.min) / scale) / Math.log(2);
354             insertQuantizerKeywords(offset, scale);
355             if (bits > 30) {
356                 bits = 30;
357                 // throw new
358                 // IllegalStateException("Cannot quantize image, noise too large");
359             }
360 
361             this.quant = new Quantizer(scale, offset);
362         }
363 
364         Cursor newPointer = this.hdr.iterator();
365         newPointer.setKey("END");
366         Cursor oldPointer = old.iterator();
367         oldPointer.setKey("BITPIX");
368 
369         copyOldKeywords(oldPointer, newPointer);
370         TileLooper tl = new TileLooper(this.imageSize, this.tileSize);
371         this.cs.initialize(parameters);
372         populateData(kern, bitpix, tl, this.cs);
373     }

375     private void copyOldKeywords(Cursor oldPointer, Cursor newPointer) {
376 
377         newPointer.add(HeaderCard.create("COMMENT"));
378         newPointer.add(HeaderCard.create("COMMENT   Header info copied from original image"));
379         newPointer.add(HeaderCard.create("COMMENT"));
380 
381         while (oldPointer.hasNext()) {
382             HeaderCard card = (HeaderCard) oldPointer.next();
383             String key = card.getKey();
384             if (key.equals("END")) {
385                 break;
386             }
387 
388             if (!TiledImageHDU.reservedKeys.contains(key)) {
389                 newPointer.add(card);
390             }
391         }
392     }

394     private CompressionScheme getCompression(String comp) {
395         if (comp == null) {
396             comp = "rice_1";
397         }
398         comp = comp.toLowerCase();
399 
400         CompressionScheme cs;
401         if (comp.equals("rice_1") || comp.equals("rice")) {
402             cs = new Rice();
403         } else if (comp.equals("gzip_1") || comp.equals("gzip")) {
404             cs = new Gzip();
405         } else if (comp.equals("hcompress_1") || comp.equals("hcompress")) {
406             cs = new HCompress();
407         } else {
408             throw new IllegalArgumentException("Unsupported compression:" + comp);
409         }
410         return cs;
411     }

413     /**
414      * Find the size and tile information in the header
415      */
416     private void getDimens(int[] axes, int[] tiles) throws FitsException {
417 
418         boolean tilesFound = true;
419         // First look for the ZTILEn keywords.
420         for (int i = 0; i < axes.length; i += 1) {
421             axes[i] = this.hdr.getIntValue("ZNAXIS" + (i + 1), -1);
422             if (axes[i] == -1) {
423                 throw new FitsException("Required ZNAXISn not found");
424             }
425             if (tilesFound) {
426                 tiles[i] = this.hdr.getIntValue("ZTILE" + (i + 1), -1);
427                 if (tiles[i] == -1) {
428                     tilesFound = false;
429                 }
430             }
431         }
432 
433         if (!tilesFound) {
434             tiles[0] = axes[0];
435             for (int i = 1; i < tiles.length; i += 1) {
436                 tiles[i] = 1;
437             }
438         }
439     }

446     public ImageHDU getImageHDU() throws FitsException, IOException {
447 
448         int[] axes = new int[this.hdr.getIntValue("ZNAXIS")];
449         this.hdr.getIntValue("ZBITPIX");
450 
451         int[] tiles = new int[axes.length];
452         getDimens(axes, tiles);
453 
454         Object data = ArrayFuncs.newInstance(this.baseClass, ArrayFuncs.reverseIndices(axes));
455         int[] dataCorner = new int[this.naxis];
456 
457         TileLooper tl = new TileLooper(axes, tiles);
458         tl.iterator();
459 
460         Object[] rows = (Object[]) getColumn("COMPRESSED_DATA");
461         String className = data.getClass().getName();
462         for (TileDescriptor td : tl) {
463             byte[] tileData = (byte[]) rows[td.count];
464             Object tile = getTile(td, tileData);
465             insertTile(tile, td.corner, td.size, data, dataCorner, this.imageSize, className, 0);
466         }
467         System.out.println("Finished the loop");
468 
469         BasicHDU bhdu = FitsFactory.hduFactory(data);
470         // importKeywords(bhdu);
471         return (ImageHDU) bhdu;
472     }

510     private Object getTile(TileDescriptor td, byte[] tileData) throws IOException {
511         int tileLen = 1;
512         int[] tsize = td.size;
513 
514         for (int element : tsize) {
515             tileLen *= element;
516         }
517 
518         tileData = this.cs.decompress(tileData, tileLen);
519         Object tile = ArrayFuncs.newInstance(this.baseClass, ArrayFuncs.reverseIndices(td.size));
520         if (this.quant == null) {
521             BufferedDataInputStream bds = new BufferedDataInputStream(new ByteArrayInputStream(tileData));
522             bds.readLArray(tile);
523         } else {
524             this.quant.fill(tileData, tile, td);
525         }
526         return tile;
527     }

529     private byte[] getTileData(TileDescriptor td, Object kern, int bitpix) {
530         int sz = Math.abs(bitpix) / 8;
531         for (int element : td.size) {
532             sz *= element;
533         }
534         int[] size = td.size;
535         int[] corn = td.corner;
536 
537         ByteArrayOutputStream bo = new ByteArrayOutputStream(sz);
538         DataOutputStream output = new DataOutputStream(bo);
539         int[] pixel = new int[size.length];
540 
541         try {
542             while (true) {
543                 writeArray(output, kern, pixel, corn, size, 0);
544                 // We'll handle the first index in writeArray so we
545                 // start at 1. Note that this indices are in FITS order,
546                 // the opposite of Java's.
547                 boolean incremented = false;
548                 for (int i = 1; i < size.length; i += 1) {
549                     if (pixel[i] < size[i] - 1) {
550                         pixel[i] += 1;
551                         incremented = true;
552                     }
553                 }
554                 if (!incremented) {
555                     break;
556                 }
557             }
558             output.close();
559             return bo.toByteArray();
560         } catch (IOException e) {
561             System.err.println("Unexpected IOException transferring data");
562             throw new RuntimeException("Unexpected exception", e);
563         }
564     }

572     /**
573      * Fill in a single tile's worth of data in the subset. Note that we are
574      * using FITS ordering of indices.
575      * 
576      * @param tileArray
577      *            The input stream containing the tile data.
578      * @param cutoutArray
579      *            The cutout array
580      * @param tileCorners
581      *            The current tile descriptor.
582      * @param corners
583      *            The corners of the cutout array within the full image
584      * @param lengths
585      *            The lengths of the cutout array
586      * @throws IOException
587      */
588     private void insertTile(Object tileData, int[] tileCorners, int tileSize[], Object cutoutData, int[] cutoutCorners, int cutoutSize[], String className, int level) {
589 
590         // Recall that our arrays are describing the cutout in
591         // the same order as FITS uses (x,y,z)
592         int x = tileCorners.length - level - 1; // Inverted index
593 
594         int txStart = 0;
595         int cxStart = 0;
596         // Does the tile start before the cutout?
597         if (tileCorners[x] < cutoutCorners[x]) {
598             txStart = cutoutCorners[x] - tileCorners[x];
599         } else {
600             cxStart = tileCorners[x] - cutoutCorners[x];
601         }
602 
603         int txCnt = tileSize[x];
604         if (tileCorners[x] + tileSize[x] > cutoutCorners[x] + cutoutSize[x]) {
605             txCnt = cutoutCorners[x] + cutoutSize[x] - tileCorners[x] - txStart;
606         }
607 
608         if (className.charAt(level + 1) == '[') {
609 
610             // We are going to recurse to the next level.
611             // Note that the arrays are in FITS order, so
612             // we need to reverse them.
613             Object[] t = (Object[]) tileData;
614             Object[] c = (Object[]) cutoutData;
615             for (int i = 0; i < txCnt; i += 1) {
616                 insertTile(t[txStart + i], tileCorners, tileSize, c[cxStart + i], cutoutCorners, cutoutSize, className, level + 1);
617             }
618 
619         } else {
620             // Just copy the data into the cutout.
621             System.arraycopy(tileData, txStart, cutoutData, cxStart, txCnt);
622         }
623     }

625     private void insertTileKeywords(Header old, CompressionScheme comp, Map<String, String> parameters, int[] axes, int[] tiles) throws FitsException {
626 
627         this.hdr.insertComment(" ");
628         this.hdr.insertComment("  Tile compression keywords ");
629         this.hdr.insertComment(" ");
630 
631         // Update the header.
632         this.hdr.addValue("ZIMAGE", true, "This is a tile compressed image");
633 
634         this.hdr.addValue("ZCMPTYPE", comp.name(), "The compression algorithm used");
635         this.hdr.addValue("ZBITPIX", old.getIntValue("BITPIX"), "The original bitpix value");
636         this.hdr.addValue("ZNAXIS", axes.length, "The original NAXIS");
637         for (int i = 0; i < axes.length; i += 1) {
638             String d = i + 1 + "";
639             this.hdr.addValue("ZNAXIS" + d, axes[i], "The original NAXIS" + d);
640             this.hdr.addValue("ZTILE" + d, tiles[i], "The tile size along this axis");
641         }
642         if (old.containsKey("SIMPLE")) {
643             this.hdr.addValue("ZSIMPLE", old.getBooleanValue("SIMPLE"), "Was primary array");
644         }
645 
646         if (old.containsKey("BLOCKED")) {
647             this.hdr.addValue("ZBLOCKED", old.getIntValue("BLOCKED"), "Old BLOCKED value");
648         }
649         if (old.containsKey("EXTEND")) {
650             this.hdr.addValue("ZEXTEND", old.getBooleanValue("EXTEND"), "Old EXTEND value");
651         }
652         if (old.containsKey("PCOUNT")) {
653             this.hdr.addValue("ZPCOUNT", old.getIntValue("PCOUNT"), "Old PCOUNT value");
654         }
655         if (old.containsKey("GCOUNT")) {
656             this.hdr.addValue("ZGCOUNT", old.getIntValue("GCOUNT"), "Old GCOUNT value");
657         }
658         if (old.containsKey("CHECKSUM")) {
659             this.hdr.addValue("ZHECKSUM", old.getStringValue("CHECKSUM"), "Old CHECKSUM value");
660         }
661         if (old.containsKey("DATASUM")) {
662             this.hdr.addValue("DATASUM", old.getStringValue("DATASUM"), "Old DATASUM value");
663         }
664         comp.updateForWrite(this.hdr, parameters);
665     }

667     private void populateData(Object kern, int bitpix, TileLooper tl, CompressionScheme cs) throws FitsException, IOException {
668 
669         getData();
670         this.deleteRows(0, 2);
671 
672         int tileCount = 0;
673 
674         Iterator<TileDescriptor> ti = tl.iterator();
675         this.kernelClass = kern.getClass().getName();
676         getData();
677 
678         while (ti.hasNext()) {
679             TileDescriptor td = ti.next();
680             for (int element : td.size) {
681             }
682             byte[] data;
683             if (this.quant == null) {
684                 data = getTileData(td, kern, bitpix);
685             } else {
686                 data = this.quant.quantize(kern, td, tileCount);
687             }
688             data = cs.compress(data);
689             addRow(new Object[]{
690                 data
691             });
692         }
693     }

695     private void writeArray(DataOutputStream output, Object data, int[] pixel, int[] corner, int[] size, int level) throws IOException {
696         char c = this.kernelClass.charAt(level + 1);
697 
698         // The indices are in FITS order, so we access them inverted.
699         int zind = size.length - level - 1;
700         int pix = corner[zind] + pixel[zind];
701         int sz = size[zind];
702 
703         switch (c) {
704         // Recurse to the next level, but pick out the appropriate sub-array.
705             case '[':
706                 writeArray(output, ((Object[]) data)[pix], pixel, corner, size, level + 1);
707                 break;
708 
709             case 'B': {
710                 byte[] temp = (byte[]) data;
711                 for (int i = pix; i < pix + sz; i += 1) {
712                     output.writeByte(temp[i]);
713                 }
714                 break;
715             }
716 
717             case 'S': {
718                 short[] temp = (short[]) data;
719                 for (int i = pix; i < pix + sz; i += 1) {
720                     output.writeShort(temp[i]);
721                 }
722                 break;
723             }
724 
725             case 'I': {
726                 int[] temp = (int[]) data;
727                 for (int i = pix; i < pix + sz; i += 1) {
728                     output.writeInt(temp[i]);
729                 }
730                 break;
731             }
732             case 'J': {
733                 long[] temp = (long[]) data;
734                 for (int i = pix; i < pix + sz; i += 1) {
735                     output.writeLong(temp[i]);
736                 }
737                 break;
738             }
739             case 'F': {
740                 float[] temp = (float[]) data;
741                 for (int i = pix; i < pix + sz; i += 1) {
742                     output.writeFloat(temp[i]);
743                 }
744                 break;
745             }
746             case 'D': {
747                 double[] temp = (double[]) data;
748                 for (int i = pix; i < pix + sz; i += 1) {
749                     output.writeDouble(temp[i]);
750                 }
751                 break;
752             }
753             default:
754                 throw new IOException("Invalid type rendering tiled image:" + this.kernelClass);
755         }
756 
757     }
