34  /**
35   * This class implements the random number generator described in the FITS tiled
36   * image compression convention, appendix A. These random numbers are used in
37   * the quantization of floating point images. Note that the discussion in that
38   * appendix assumes one-based arrays rather than Java's zero-based arrays so
39   * that the formulae specified need to be adjusted. In typical usage the
40   * computeOffset(int) call will be invoked at the beginning of each tile with
41   * the tile index as an argument (first tile uses 0, the next uses 1 and so
42   * forth). Then next() is called to get the dither for each pixel in the tile.
43   * Note that these numbers range from -0.5 to 0.5 rather than 0 to 1 since so
44   * that the subtraction by 0.5 in the reference is not required.
45   * 
46   * @author tmcglynn
47   */
48  public class QuantizeRandoms {
49  
50      /** The set of 10,000 random numbers used */
51      private double[] values;
52  
53      /** The last index requested */
54      private int nextIndex = -1;
55  
56      /** The last starting index used. */
57      private int lastStart = -1;
58  
59      /** Have the values been initialized? */
60      private boolean ready = false;
61  
62      /** The number of values to be generated */
63      private final int NVAL = 10000;
64  
65      /**
66       * The multiplier we use when trying to get a randomish starting location in
67       * the array.
68       */
69      private final int MULT = 500;
70  
71      /**
72       * Generally we try to start at a random location in the first MULT entries
73       * within the array using an integer we increment for each new tile.
74       * location in the array.
75       */
76      public void computeOffset(int n) {
77          if (!this.ready) {
78              initialize();
79          }
80          while (n < 0) {
81              n += this.NVAL;
82          }
83          while (n >= this.NVAL) {
84              n -= this.NVAL;
85          }
86          this.nextIndex = (int) (this.MULT * (this.values[n] + 0.5));
87      }
88  
89      /** Initialize the sequence of NVAL random numbers */
90      private void initialize() {
91  
92          this.values = new double[this.NVAL];
93  
94          double a = 16807;
95          double m = 2147483647;
96          double seed = 1;
97          double temp;
98  
99          for (int ii = 0; ii < this.NVAL; ii += 1) {
100             temp = a * seed;
101             seed = temp - m * Math.floor(temp / m);
102             this.values[ii] = seed / m - 0.5;
103         }
104         this.ready = true;
105         if (seed != 1043618065) {
106             throw new IllegalStateException("Final seed has unexpected value");
107         }
108     }
109 
110     /**
111      * Get the next number in the fixed sequence. This may be called any number
112      * of times between calls to computeOffset(). If it is called before the
113      * first call to computeOffset(), then computOffset(0) is called to get the
114      * initial index offset.
115      */
116     public double next() {
117 
118         if (this.lastStart < 0) {
119             computeOffset(0);
120             this.lastStart = 0;
121         }
122         if (this.nextIndex >= this.NVAL) {
123             this.lastStart += 1;
124             if (this.lastStart >= this.NVAL) {
125                 this.lastStart = 0;
126             }
127             computeOffset(this.lastStart);
128         }
129         int currIndex = this.nextIndex;
130         this.nextIndex += 1;
131         return this.values[currIndex];
132     }
133 }
