25  import org.junit.Test;
26  import static org.junit.Assert.assertEquals;
27  import junit.framework.JUnit4TestAdapter;
28  import nom.tam.util.ArrayFuncs;
29  
30  public class ArrayFuncs2Test {
31  
32      /**
33       * Test and demonstrate the ArrayFuncs methods.
34       */
35      @Test
36      public void test() {
37  
38          int[][][] test1 = new int[10][9][8];
39          boolean[][] test2 = new boolean[4][];
40          test2[0] = new boolean[5];
41          test2[1] = new boolean[4];
42          test2[2] = new boolean[3];
43          test2[3] = new boolean[2];
44  
45          double[][] test3 = new double[10][20];
46          StringBuffer[][] test4 = new StringBuffer[3][2];
47  
48          assertEquals("getBaseClass()", int.class, ArrayFuncs.getBaseClass(test1));
49          assertEquals("getBaseLength()", 4, ArrayFuncs.getBaseLength(test1));
50          assertEquals("computeSize()", 4 * 8 * 9 * 10, ArrayFuncs.computeSize(test1));
51          assertEquals("computeLSize()", 4 * 8 * 9 * 10L, ArrayFuncs.computeLSize(test1));
52  
53          assertEquals("getBaseClass(boolean)", boolean.class, ArrayFuncs.getBaseClass(test2));
54          assertEquals("getBaseLength(boolean)", 1, ArrayFuncs.getBaseLength(test2));
55          assertEquals("computeSize() not rect", 1 * (2 + 3 + 4 + 5), ArrayFuncs.computeSize(test2));
56          assertEquals("computeLSize() not rect", 1L * (2 + 3 + 4 + 5), ArrayFuncs.computeLSize(test2));
57  
58          assertEquals("getBaseClass(double)", double.class, ArrayFuncs.getBaseClass(test3));
59          assertEquals("getBaseLength(double)", 8, ArrayFuncs.getBaseLength(test3));
60          assertEquals("computeSize(double)", 8 * 10 * 20, ArrayFuncs.computeSize(test3));
61          assertEquals("computeLSize(double)", 8 * 10 * 20L, ArrayFuncs.computeLSize(test3));
62  
63          assertEquals("getBaseClass(StrBuf)", StringBuffer.class, ArrayFuncs.getBaseClass(test4));
64          assertEquals("getBaseLength(StrBuf)", -1, ArrayFuncs.getBaseLength(test4));
65          assertEquals("computeSize(StrBuf)", 0, ArrayFuncs.computeSize(test4));
66          assertEquals("computeLSize(StrBuf)", 0L, ArrayFuncs.computeLSize(test4));
67  
68          Object[] agg = new Object[4];
69          agg[0] = test1;
70          agg[1] = test2;
71          agg[2] = test3;
72          agg[3] = test4;
73  
74          assertEquals("getBaseClass(Object[])", Object.class, ArrayFuncs.getBaseClass(agg));
75          assertEquals("getBaseLength(Object[])", -1, ArrayFuncs.getBaseLength(agg));
76  
77          // Add up all the primitive arrays and ignore the objects.
78          assertEquals("computeSize(Object[])", 2880 + 14 + 1600 + 0, ArrayFuncs.computeSize(agg));
79          assertEquals("computeLSize(Object[])", 2880L + 14 + 1600 + 0, ArrayFuncs.computeLSize(agg));
80  
81          // Try allocating a very large array. This is likely to fail
82          // in the allocation step, so don't consider that to be a failure.
83          try {
84              float[][] data = new float[10000][30000];
85              long expect = 10000L * 30000 * 4;
86              assertEquals("computLSize(big)", ArrayFuncs.computeLSize(data), expect);
87  
88          } catch (Error e) {
89              System.out.println("Unable to allocate large array. Test skipped");
90          }
91  
92          for (int i = 0; i < test1.length; i += 1) {
93              for (int j = 0; j < test1[i].length; j += 1) {
94                  for (int k = 0; k < test1[i][j].length; k += 1) {
95                      test1[i][j][k] = i + j + k;
96                  }
97              }
98          }
99          int[][][] test5 = (int[][][]) ArrayFuncs.deepClone(test1);
100 
101         assertEquals("deepClone()", true, ArrayFuncs.arrayEquals(test1, test5));
102         test5[1][1][1] = -3;
103         assertEquals("arrayEquals()", false, ArrayFuncs.arrayEquals(test1, test5));
104 
105         int[] dimsOrig = ArrayFuncs.getDimensions(test1);
106         int[] test6 = (int[]) ArrayFuncs.flatten(test1);
107 
108         int[] dims = ArrayFuncs.getDimensions(test6);
109 
110         assertEquals("getDimensions()", 3, dimsOrig.length);
111         assertEquals("getDimensions()", 10, dimsOrig[0]);
112         assertEquals("getDimensions()", 9, dimsOrig[1]);
113         assertEquals("getDimensions()", 8, dimsOrig[2]);
114         assertEquals("flatten()", 1, dims.length);
115 
116         int[] newdims = {
117             8,
118             9,
119             10
120         };
121 
122         int[][][] test7 = (int[][][]) ArrayFuncs.curl(test6, newdims);
123 
124         int[] dimsAfter = ArrayFuncs.getDimensions(test7);
125 
126         assertEquals("curl()", 3, dimsAfter.length);
127         assertEquals("getDimensions()", 8, dimsAfter[0]);
128         assertEquals("getDimensions()", 9, dimsAfter[1]);
129         assertEquals("getDimensions()", 10, dimsAfter[2]);
130 
131         byte[][][] xtest1 = (byte[][][]) ArrayFuncs.convertArray(test1, byte.class);
132 
133         assertEquals("convertArray(toByte)", byte.class, ArrayFuncs.getBaseClass(xtest1));
134         assertEquals("convertArray(tobyte)", test1[3][3][3], (int) xtest1[3][3][3]);
135 
136         double[][][] xtest2 = (double[][][]) ArrayFuncs.convertArray(test1, double.class);
137         assertEquals("convertArray(toByte)", double.class, ArrayFuncs.getBaseClass(xtest2));
138         assertEquals("convertArray(tobyte)", test1[3][3][3], (int) xtest2[3][3][3]);
139 
140         int[] xtest3 = (int[]) ArrayFuncs.newInstance(int.class, 20);
141         int[] xtd = ArrayFuncs.getDimensions(xtest3);
142         assertEquals("newInstance(vector)", 1, xtd.length);
143         assertEquals("newInstance(vector)", 20, xtd[0]);
144         double[][][][] xtest4 = (double[][][][]) ArrayFuncs.newInstance(double.class, new int[]{
145             5,
146             4,
147             3,
148             2
149         });
150         xtd = ArrayFuncs.getDimensions(xtest4);
151         assertEquals("newInstance(tensor)", 4, xtd.length);
152         assertEquals("newInstance(tensor)", 5, xtd[0]);
153         assertEquals("newInstance(tensor)", 4, xtd[1]);
154         assertEquals("newInstance(tensor)", 3, xtd[2]);
155         assertEquals("newInstance(tensor)", 2, xtd[3]);
156         assertEquals("nElements()", 120, ArrayFuncs.nElements(xtest4));
157         assertEquals("nLElements()", 120L, ArrayFuncs.nLElements(xtest4));
158 
159         ArrayFuncs.testPattern(xtest4, (byte) -1);
160 
161         assertEquals("testPattern()", (double) -1, xtest4[0][0][0][0], 0);
162         assertEquals("testPattern()", (double) 118, xtest4[4][3][2][1], 0);
163         double[] xtest4x = (double[]) ArrayFuncs.getBaseArray(xtest4);
164 
165         assertEquals("getBaseArray()", 2, xtest4x.length);
166 
167         double[] x = {
168             1,
169             2,
170             3,
171             4,
172             5
173         };
174         double[] y = new double[x.length];
175         for (int i = 0; i < y.length; i += 1) {
176             y[i] = x[i] + 1.E-10;
177         }
178 
179         assertEquals("eqTest", false, ArrayFuncs.arrayEquals(x, y));
180         assertEquals("eqTest2", true, ArrayFuncs.arrayEquals(x, y, 0., 1.e-9));
181         assertEquals("eqTest3", true, ArrayFuncs.arrayEquals(x, y, 1.E-5, 1.e-9));
182         assertEquals("eqTest4", false, ArrayFuncs.arrayEquals(x, y, 0., 1.e-11));
183         assertEquals("eqTest5", false, ArrayFuncs.arrayEquals(x, y, 1.E-5, 0.));
184 
185         float[] fx = {
186             1,
187             2,
188             3,
189             4,
190             5
191         };
192         float[] fy = new float[fx.length];
193         for (int i = 0; i < fy.length; i += 1) {
194             fy[i] = fx[i] + 1.E-5F;
195         }
196 
197         assertEquals("eqTest6", false, ArrayFuncs.arrayEquals(fx, fy));
198         assertEquals("eqTest7", true, ArrayFuncs.arrayEquals(fx, fy, 1.E-4, 0.));
199         assertEquals("eqTest8", false, ArrayFuncs.arrayEquals(fx, fy, 1.E-6, 0.));
200         assertEquals("eqTest9", false, ArrayFuncs.arrayEquals(fx, fy, 0., 0.));
201         assertEquals("eqTest10", false, ArrayFuncs.arrayEquals(fx, fy, 0., 1.E-4));
202 
203     }
