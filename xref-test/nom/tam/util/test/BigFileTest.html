25  import org.junit.Test;
26  import static org.junit.Assert.assertEquals;
27  import junit.framework.JUnit4TestAdapter;
28  
29  import nom.tam.util.BufferedFile;
30  import nom.tam.util.BufferedDataInputStream;
31  import java.io.FileInputStream;
32  
33  public class BigFileTest {
34  
35      @Test
36      public void test() throws Exception {
37          try {
38              // First create a 3 GB file.
39              String fname = System.getenv("BIGFILETEST");
40              if (fname == null) {
41                  System.out.println("BIGFILETEST environment not set.  Returning without test");
42                  return;
43              }
44              System.out.println("Big file test.  Takes quite a while.");
45              byte[] buf = new byte[100000000]; // 100 MB
46              BufferedFile bf = new BufferedFile(fname, "rw");
47              byte sample = 13;
48  
49              for (int i = 0; i < 30; i += 1) {
50                  bf.write(buf); // 30 x 100 MB = 3 GB.
51                  if (i == 24) {
52                      bf.write(new byte[]{
53                          sample
54                      });
55                  } // Add a marker.
56              }
57              bf.close();
58  
59              // Now try to skip within the file.
60              bf = new BufferedFile(fname, "r");
61              long skip = 2500000000L; // 2.5 G
62  
63              long val1 = bf.skipBytes(skip);
64              long val2 = bf.getFilePointer();
65              int val = bf.read();
66              bf.close();
67  
68              assertEquals("SkipResult", skip, val1);
69              assertEquals("SkipPos", skip, val2);
70              assertEquals("SkipVal", (int) sample, val);
71  
72              BufferedDataInputStream bdis = new BufferedDataInputStream(new FileInputStream(fname));
73              val1 = bdis.skipBytes(skip);
74              val = bdis.read();
75              bdis.close();
76              assertEquals("SSkipResult", skip, val1);
77              assertEquals("SSkipVal", (int) sample, val);
78          } catch (Exception e) {
79              e.printStackTrace(System.err);
80              throw e;
81          }
82      }
83  }
