84      public static void main1(String[] args) throws Exception {
85          int[] test = new int[100];
86          for (int i = 0; i < test.length; i += 1) {
87              if (i % 2 != 0) {
88                  test[i] = 1000 - 2 * i;
89              } else {
90                  test[i] = 1000 + 2 * i;
91              }
92          }
93          Rice comp = new Rice();
94          Map<String, String> init = new HashMap<String, String>();
95          init.put("bitpix", "32");
96          init.put("block", "32");
97          init.put("length", "100");
98          comp.initialize(init);
99  
100         ByteArrayOutputStream bo = new ByteArrayOutputStream();
101         BufferedDataOutputStream d = new BufferedDataOutputStream(bo);
102         d.write(test);
103         d.close();
104         byte[] input = bo.toByteArray();
105         byte[] result = comp.compress(input);
106         System.out.println("Result len:" + result.length);
107         for (int i = 0; i < result.length; i += 1) {
108             System.out.printf("%d: %3d %2x\n", i, result[i], result[i]);
109         }
110 
111         result = comp.decompress(result, 100);
112         DataInputStream bi = new DataInputStream(new ByteArrayInputStream(result));
113         for (int i = 0; i < 100; i += 1) {
114             System.out.println(i + ": " + bi.readInt());
115         }
116     }
