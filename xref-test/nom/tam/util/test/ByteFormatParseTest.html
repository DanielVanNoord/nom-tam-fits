46      @Test
47      public void testInt() throws Exception {
48  
49          for (int i = 0; i < 10; i += 1) {
50              buffer[i] = (byte) ' ';
51          }
52          bp.setOffset(0);
53          assertEquals("IntBlank", 0, bp.getInt(10));
54  
55          bf.setAlign(true);
56          bf.setTruncationThrow(false);
57  
58          int[] tint = new int[100];
59  
60          tint[0] = Integer.MIN_VALUE;
61          tint[1] = Integer.MAX_VALUE;
62          tint[2] = 0;
63  
64          for (int i = 0; i < tint.length; i += 1) {
65              tint[i] = (int) (Integer.MAX_VALUE * (2 * (Math.random() - .5)));
66          }
67  
68          // Write 100 numbers
69          int colSize = 12;
70          while (cnt < tint.length) {
71              offset = bf.format(tint[cnt], buffer, offset, colSize);
72              cnt += 1;
73              if (cnt % 8 == 0) {
74                  offset = bf.format("\n", buffer, offset, 1);
75              }
76          }
77  
78          // Now see if we can get them back
79          bp.setOffset(0);
80          for (int i = 0; i < tint.length; i += 1) {
81  
82              int chk = bp.getInt(colSize);
83  
84              assertEquals("IntegersRA", chk, tint[i]);
85              if ((i + 1) % 8 == 0) {
86                  bp.skip(1);
87              }
88          }
89  
90          // Now do it with left-aligned numbers.
91          bf.setAlign(false);
92          bp.setFillFields(true);
93          offset = 0;
94          colSize = 12;
95          cnt = 0;
96          offset = 0;
97          while (cnt < tint.length) {
98              int oldOffset = offset;
99              offset = bf.format(tint[cnt], buffer, offset, colSize);
100             int nb = colSize - (offset - oldOffset);
101             if (nb > 0) {
102                 offset = bf.alignFill(buffer, offset, nb);
103             }
104             cnt += 1;
105             if (cnt % 8 == 0) {
106                 offset = bf.format("\n", buffer, offset, 1);
107             }
108         }
109 
110         // Now see if we can get them back
111         bp.setOffset(0);
112         for (int i = 0; i < tint.length; i += 1) {
113 
114             int chk = bp.getInt(colSize);
115 
116             assertEquals("IntegersLA", chk, tint[i]);
117             if ((i + 1) % 8 == 0) {
118                 bp.skip(1);
119             }
120         }
121 
122         offset = 0;
123         colSize = 12;
124         cnt = 0;
125         offset = 0;
126         while (cnt < tint.length) {
127             offset = bf.format(tint[cnt], buffer, offset, colSize);
128             cnt += 1;
129             if (cnt % 8 == 0) {
130                 offset = bf.format("\n", buffer, offset, 1);
131             }
132         }
133 
134         String myStr = new String(buffer, 0, offset);
135         assertEquals("No spaces", -1, myStr.indexOf(" "));
136 
137         bf.setAlign(false);
138 
139         offset = 0;
140         colSize = 12;
141         cnt = 0;
142         offset = 0;
143         while (cnt < tint.length) {
144             offset = bf.format(tint[cnt], buffer, offset, colSize);
145             offset = bf.format(" ", buffer, offset, 1);
146             cnt += 1;
147         }
148         myStr = new String(buffer, 0, offset);
149         String[] array = myStr.split(" ");
150 
151         assertEquals("Split size", 100, array.length);
152 
153         for (int i = 0; i < array.length; i += 1) {
154             assertEquals("Parse token", tint[i], Integer.parseInt(array[i]));
155         }
156 
157         bf.setTruncationThrow(false);
158 
159         int val = 1;
160         Arrays.fill(buffer, (byte) ' ');
161 
162         for (int i = 0; i < 10; i += 1) {
163             offset = bf.format(val, buffer, 0, 6);
164             String test = (val + "      ").substring(0, 6);
165             if (i < 6) {
166                 assertEquals("TestTrunc" + i, test, new String(buffer, 0, 6));
167             } else {
168                 assertEquals("TestTrunc" + i, "******", new String(buffer, 0, 6));
169             }
170             val *= 10;
171         }
172 
173         bf.setTruncationThrow(true);
174         val = 1;
175         for (int i = 0; i < 10; i += 1) {
176             boolean thrown = false;
177             try {
178                 offset = bf.format(val, buffer, 0, 6);
179             } catch (TruncationException e) {
180                 thrown = true;
181             }
182             if (i < 6) {
183                 assertEquals("TestTruncThrow" + i, false, thrown);
184             } else {
185                 assertEquals("TestTruncThrow" + i, true, thrown);
186             }
187             val *= 10;
188         }
189     }

230     @Test
231     public void testFloat() throws Exception {
232 
233         for (int i = 0; i < 10; i += 1) {
234             buffer[i] = (byte) ' ';
235         }
236         bp.setOffset(0);
237         assertEquals("FloatBlank", 0.f, bp.getFloat(10), 0.);
238 
239         float[] flt = new float[100];
240         for (int i = 6; i < flt.length; i += 1) {
241             flt[i] = (float) (2 * (Math.random() - 0.5) * Math.pow(10, 60 * (Math.random() - 0.5)));
242         }
243 
244         flt[0] = Float.MAX_VALUE;
245         flt[1] = Float.MIN_VALUE;
246         flt[2] = 0;
247         flt[3] = Float.NaN;
248         flt[4] = Float.POSITIVE_INFINITY;
249         flt[5] = Float.NEGATIVE_INFINITY;
250 
251         bf.setTruncationThrow(false);
252         bf.setAlign(true);
253 
254         offset = 0;
255         cnt = 0;
256 
257         while (cnt < flt.length) {
258             offset = bf.format(flt[cnt], buffer, offset, 24);
259             cnt += 1;
260             if (cnt % 4 == 0) {
261                 offset = bf.format("\n", buffer, offset, 1);
262             }
263         }
264 
265         bp.setOffset(0);
266 
267         for (int i = 0; i < flt.length; i += 1) {
268 
269             float chk = bp.getFloat(24);
270 
271             float dx = Math.abs(chk - flt[i]);
272             if (flt[i] != 0) {
273                 dx = dx / Math.abs(flt[i]);
274             }
275             if (Float.isNaN(flt[i])) {
276                 assertEquals("Float check:" + i, true, Float.isNaN(chk));
277             } else if (Float.isInfinite(flt[i])) {
278                 assertEquals("Float check:" + i, flt[i], chk, 0);
279             } else {
280                 assertEquals("Float check:" + i, 0., dx, 1.e-6);
281             }
282             if ((i + 1) % 4 == 0) {
283                 bp.skip(1);
284             }
285         }
286     }

363     @Test
364     public void testString() throws Exception {
365 
366         offset = 0;
367         String bigStr = "abcdefghijklmnopqrstuvwxyz";
368 
369         for (int i = 0; i < 100; i += 1) {
370             offset = bf.format(bigStr.substring(i % 27), buffer, offset, 13);
371             offset = bf.format(" ", buffer, offset, 1);
372         }
373 
374         bp.setOffset(0);
375         for (int i = 0; i < 100; i += 1) {
376             int ind = i % 27;
377             if (ind > 13) {
378                 ind = 13;
379             }
380             String want = bigStr.substring(i % 27);
381             if (want.length() > 13) {
382                 want = want.substring(0, 13);
383             }
384             String s = bp.getString(want.length());
385             assertEquals("String:" + i, want, s);
386             bp.skip(1);
387         }
