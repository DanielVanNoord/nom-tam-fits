106     /**
107      * This is an assertion for a special case when a loss full algorithm is
108      * used and a quantification with null checks. The will result in wrong null
109      * values, because some of the values representing blank will be lost.
110      */
111     private void assertArrayEquals(double[] expected, double[] actual, double delta) {
112         Assert.assertEquals(expected.length, actual.length);
113         for (int index = 0; index < actual.length; index++) {
114             double d1 = expected[index];
115             double d2 = actual[index];
116             if (Double.isNaN(d1) || Double.isNaN(d2)) {
117                 Assert.assertTrue(true); // ;-)
118             } else {
119                 Assert.assertEquals(d1, d2, delta);
120             }
121         }
122     }

135     private <T> void assertCompressedToUncompressedImage(String fileName, String unCompfileName, Class<T> clazz, IHDUAsserter<T> reader) throws Exception {
136         try (Fits f = new Fits(fileName); Fits unFits = new Fits(unCompfileName)) {
137             ImageHDU hdu = readHDU(unFits, ImageHDU.class);
138             unFits.deleteHDU(0);
139             CompressedImageHDU uncompHdu = readHDU(f, CompressedImageHDU.class);
140             f.deleteHDU(0);
141             while (hdu != null && uncompHdu != null) {
142                 T compressedData = (T) uncompHdu.asImageHDU().getData().getData();
143                 T orgData = (T) hdu.getData().getData();
144                 // show travis something is going on
145                 System.out.println("Asserting image data! " + (assertionCounter++));
146                 reader.assertData(orgData, compressedData);
147                 hdu = readHDU(unFits, ImageHDU.class);
148                 if (hdu != null) {
149                     unFits.deleteHDU(0);
150                 }
151                 uncompHdu = readHDU(f, CompressedImageHDU.class);
152                 if (uncompHdu != null) {
153                     f.deleteHDU(0);
154                 }
155             }
156             Assert.assertFalse(hdu != null || uncompHdu != null);
157         }
158     }

176     private void assertFloatImage(FloatBuffer result, float[][] expected, float delta) {
177         float[] real = new float[expected[0].length];
178         for (float[] expectedPart : expected) {
179             result.get(real);
180             Assert.assertEquals(expectedPart.length, real.length);
181             for (int subindex = 0; subindex < expectedPart.length; subindex++) {
182                 float expectedFloat = expectedPart[subindex];
183                 float realFloat = real[subindex];
184                 if (!Float.isNaN(expectedFloat) && !Float.isNaN(realFloat)) {
185                     Assert.assertEquals(expectedFloat, realFloat, delta);
186                 }
187             }
188         }
189     }

333     private void dispayImage(short[][] data) {
334         JFrame frame = new JFrame("FrameDemo");
335         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
336         frame.pack();
337         frame.setVisible(true);
338 
339         BufferedImage img = (BufferedImage) frame.createImage(300, 300);
340 
341         Graphics2D g = img.createGraphics(); // Get a Graphics2D object
342 
343         for (int y = 0; y < 300; y++) {
344             for (int x = 0; x < 300; x++) {
345                 float grayScale = data[x][y] / 4000f;
346                 g.setColor(new Color(grayScale, grayScale, grayScale));
347                 g.drawRect(x, y, 1, 1);
348             }
349         }
350         ImageIcon icon = new ImageIcon(img);
351         JLabel label = new JLabel(icon);
352         frame.getContentPane().add(label, BorderLayout.CENTER);
353         frame.revalidate();
354         frame.repaint();
355         frame.pack();
356         frame.setVisible(true);
357     }

359     private HeaderCard findCompressOption(Header header, String key) {
360         int index = 1;
361         while (true) {
362             HeaderCard card = header.findCard(Compression.ZNAMEn.n(index));
363             if (card.getValue().equals(key)) {
364                 return header.findCard(Compression.ZVALn.n(index));
365             }
366             index++;
367         }
368     }

370     private boolean isEmptyImage(BasicHDU<?> result) {
371         return result instanceof ImageHDU && ((ImageHDU) result).getData().getData() == null;
372     }

374     private Object readAll(String fileName, int index) throws Exception {
375         try (Fits f = new Fits(fileName)) {
376             BasicHDU<?> hdu = f.getHDU(index);
377             if (hdu instanceof CompressedImageHDU) {
378                 CompressedImageHDU bhdu = (CompressedImageHDU) hdu;
379                 return bhdu.getUncompressedData();
380             }
381             if (hdu instanceof ImageHDU) {
382                 ImageHDU bhdu = (ImageHDU) hdu;
383                 return bhdu.getData().getData();
384             }
385         }
386         return null;
387     }
388 
389     private ImageHDU readCompressedHdu(String fileName, int index) throws Exception {
390         try (Fits f = new Fits(fileName)) {
391             BasicHDU<?> hdu = f.getHDU(index);
392             if (hdu instanceof CompressedImageHDU) {
393                 CompressedImageHDU bhdu = (CompressedImageHDU) hdu;
394                 return bhdu.asImageHDU();
395             }
396         }
397         return null;
398     }

423     private <T> T readHDU(Fits fits, Class<T> clazz) throws FitsException, IOException {
424         BasicHDU<?> result = fits.readHDU();
425         while (result != null && (isEmptyImage(result) || !clazz.isAssignableFrom(result.getClass()))) {
426             result = fits.readHDU();
427         }
428         return (T) result;
429     }
