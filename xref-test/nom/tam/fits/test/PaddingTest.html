42  /**
43   * Test that we can read files that fail due to lack of padding in the final
44   * HDU.
45   */
46  public class PaddingTest {
47  
48      @Test
49      public void test1() throws Exception {
50  
51          Fits f = new Fits();
52  
53          byte[][] bimg = new byte[20][20];
54          for (int i = 0; i < 20; i += 1) {
55              for (int j = 0; j < 20; j += 1) {
56                  bimg[i][j] = (byte) (i + j);
57              }
58          }
59  
60          BasicHDU hdu = Fits.makeHDU(bimg);
61          Header hdr = hdu.getHeader();
62          hdr.addValue("NEWKEY", "TESTVALUE", "Test keyword");
63          BufferedFile bf = new BufferedFile("target/padding1.fits", "rw");
64          hdr.write(bf);
65          bf.writeArray(bimg); // The data but no following padding.
66          bf.flush();
67          bf.close();
68  
69          // Now try reading this back.
70          f = new Fits("target/padding1.fits");
71  
72          try {
73              f.read();
74          } catch (PaddingException e) {
75              assertEquals("HDUCount", 0, f.getNumberOfHDUs());
76              f.addHDU(e.getTruncatedHDU());
77              assertEquals("HDUCount2", 1, f.getNumberOfHDUs());
78          }
79  
80          ImageHDU hdu0 = (ImageHDU) (f.getHDU(0));
81          byte[][] aa = (byte[][]) hdu0.getKernel();
82          int miss = 0;
83          int match = 0;
84          for (int i = 0; i < 20; i += 1) {
85              for (int j = 0; j < 20; j += 1) {
86                  if (aa[i][j] != (byte) (i + j)) {
87                      miss += 1;
88                  } else {
89                      match += 1;
90                  }
91              }
92          }
93          assertEquals("PadMiss1:", miss, 0);
94          assertEquals("PadMatch1:", match, 400);
95          // Make sure we got the real header and not the one generated strictly
96          // from the data.
97          assertEquals("Update header:", hdu0.getHeader().getStringValue("NEWKEY"), "TESTVALUE");
98  
99          nom.tam.image.StandardImageTiler it = hdu0.getTiler();
100 
101         // Remember that the tile is always a flattened
102         // 1-D representation of the data.
103         byte[] data = (byte[]) it.getTile(new int[]{
104             2,
105             2
106         }, new int[]{
107             2,
108             2
109         });
110 
111         assertEquals("tilet1:", data.length, 4);
112         assertEquals("tilet2:", data[0] + 0, 4);
113         assertEquals("tilet3:", data[1] + 0, 5);
114         assertEquals("tilet4:", data[2] + 0, 5);
115         assertEquals("tilet5:", data[3] + 0, 6);
116 
117     }

119     @Test
120     public void test2() throws Exception {
121 
122         Fits f = new Fits();
123 
124         byte[][] bimg = new byte[20][20];
125         for (int i = 0; i < 20; i += 1) {
126             for (int j = 0; j < 20; j += 1) {
127                 bimg[i][j] = (byte) (i + j);
128             }
129         }
130 
131         BasicHDU hdu = Fits.makeHDU(bimg);
132         f.addHDU(hdu);
133 
134         // First create a FITS file with a truncated second HDU.
135         BufferedFile bf = new BufferedFile("target/padding2.fits", "rw");
136         f.write(bf);
137 
138         hdu.getHeader().setXtension("IMAGE");
139         Cursor curs = hdu.getHeader().iterator();
140         int cnt = 0;
141         // Write the header
142         while (curs.hasNext()) {
143             bf.write(((HeaderCard) curs.next()).toString().getBytes());
144             cnt += 1;
145         }
146 
147         // The padding between header and data
148         byte[] b = new byte[(36 - cnt) * 80]; // Assuming fewer than 36 cards.
149         for (int i = 0; i < b.length; i += 1) {
150             b[i] = 32; // i.e., a blank
151         }
152         bf.write(b);
153         for (int i = 0; i < 20; i += 1) {
154             for (int j = 0; j < 20; j += 1) {
155                 bimg[i][j] = (byte) (2 * (i + j));
156             }
157         }
158         bf.writeArray(bimg); // The data but no following padding.
159         bf.flush();
160         bf.close();
161 
162         // Now try reading this back.
163         f = new Fits("target/padding2.fits");
164 
165         try {
166             f.read();
167         } catch (PaddingException e) {
168             assertEquals("HDUCount", 1, f.getNumberOfHDUs());
169             f.addHDU(e.getTruncatedHDU());
170             assertEquals("HDUCount2", 2, f.getNumberOfHDUs());
171         }
172 
173         ImageHDU hdu0 = (ImageHDU) (f.getHDU(0));
174         ImageHDU hdu1 = (ImageHDU) (f.getHDU(1));
175         byte[][] aa = (byte[][]) hdu0.getKernel();
176         byte[][] bb = (byte[][]) hdu1.getKernel();
177         int miss = 0;
178         int match = 0;
179         for (int i = 0; i < 20; i += 1) {
180             for (int j = 0; j < 20; j += 1) {
181                 if (bb[i][j] != (byte) (2 * aa[i][j])) {
182                     miss += 1;
183                 } else {
184                     match += 1;
185                 }
186             }
187         }
188         assertEquals("PadMiss2:", miss, 0);
189         assertEquals("PadMatch2:", match, 400);
190     }
