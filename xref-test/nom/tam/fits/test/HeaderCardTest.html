35      @Test
36      public void test1() throws Exception {
37  
38          HeaderCard p;
39          p = new HeaderCard("SIMPLE  =                     T");
40  
41          assertEquals("t1", "SIMPLE", p.getKey());
42          assertEquals("t2", "T", p.getValue());
43          assertNull("t3", p.getComment());
44  
45          p = new HeaderCard("VALUE   =                   123");
46          assertEquals("t4", "VALUE", p.getKey());
47          assertEquals("t5", "123", p.getValue());
48          assertNull("t3", p.getComment());
49  
50          p = new HeaderCard("VALUE   =    1.23698789798798E23 / Comment ");
51          assertEquals("t6", "VALUE", p.getKey());
52          assertEquals("t7", "1.23698789798798E23", p.getValue());
53          assertEquals("t8", "Comment", p.getComment());
54  
55          String lng = "111111111111111111111111111111111111111111111111111111111111111111111111";
56          p = new HeaderCard("COMMENT " + lng);
57          assertEquals("t9", "COMMENT", p.getKey());
58          assertNull("t10", p.getValue());
59          assertEquals("t11", lng, p.getComment());
60  
61          boolean thrown = false;
62          try {
63              //
64              p = new HeaderCard("VALUE   = '   ");
65          } catch (Exception e) {
66              thrown = true;
67          }
68          assertEquals("t12", true, thrown);
69  
70          p = new HeaderCard("COMMENT " + lng + lng);
71          assertEquals("t13", lng, p.getComment());
72  
73          HeaderCard z = new HeaderCard("TTTT", 1.234567891234567891234567e101, "a comment");
74          assertTrue("t14", z.toString().indexOf("E") > 0);
75      }

77      @Test
78      public void test3() throws Exception {
79  
80          HeaderCard p = new HeaderCard("KEY", "VALUE", "COMMENT");
81          assertEquals("x1", "KEY     = 'VALUE   '           / COMMENT                                        ", p.toString());
82  
83          p = new HeaderCard("KEY", 123, "COMMENT");
84          assertEquals("x2", "KEY     =                  123 / COMMENT                                        ", p.toString());
85          p = new HeaderCard("KEY", 1.23, "COMMENT");
86          assertEquals("x3", "KEY     =                 1.23 / COMMENT                                        ", p.toString());
87          p = new HeaderCard("KEY", true, "COMMENT");
88          assertEquals("x4", "KEY     =                    T / COMMENT                                        ", p.toString());
89  
90          boolean thrown = false;
91          try {
92              p = new HeaderCard("LONGKEYWORD", 123, "COMMENT");
93          } catch (Exception e) {
94              thrown = true;
95          }
96          assertEquals("x5", true, thrown);
97  
98          thrown = false;
99          String lng = "00000000001111111111222222222233333333334444444444555555555566666666667777777777";
100         try {
101             p = new HeaderCard("KEY", lng, "COMMENT");
102         } catch (Exception e) {
103             thrown = true;
104         }
105         assertEquals("x6", true, thrown);
106 
107         // Only trailing spaces are stripped.
108         p = new HeaderCard("STRING", "VALUE", null);
109         assertEquals("x6", "VALUE", p.getValue());
110 
111         p = new HeaderCard("STRING", "VALUE ", null);
112         assertEquals("x7", "VALUE", p.getValue());
113 
114         p = new HeaderCard("STRING", " VALUE", null);
115         assertEquals("x8", " VALUE", p.getValue());
116 
117         p = new HeaderCard("STRING", " VALUE ", null);
118         assertEquals("x9", " VALUE", p.getValue());
119 
120         p = new HeaderCard("QUOTES", "ABC'DEF", null);
121         assertEquals("x10", "ABC'DEF", p.getValue());
122         assertEquals("x10b", p.toString().indexOf("''") > 0, true);
123 
124         p = new HeaderCard("QUOTES", "ABC''DEF", null);
125         assertEquals("x11", "ABC''DEF", p.getValue());
126         assertEquals("x10b", p.toString().indexOf("''''") > 0, true);
127     }

129     @Test
130     public void testHierarch() throws Exception {
131 
132         HeaderCard hc;
133         String key = "HIERARCH.TEST1.TEST2.INT";
134         boolean thrown = false;
135         try {
136             hc = new HeaderCard(key, 123, "Comment");
137         } catch (Exception e) {
138             thrown = true;
139         }
140         assertEquals("h1", true, thrown);
141 
142         String card = "HIERARCH TEST1 TEST2 INT=           123 / Comment                               ";
143         hc = new HeaderCard(card);
144         assertEquals("h2", "HIERARCH", hc.getKey());
145         assertNull("h3", hc.getValue());
146         assertEquals("h4", "TEST1 TEST2 INT=           123 / Comment", hc.getComment());
147 
148         FitsFactory.setUseHierarch(true);
149 
150         hc = new HeaderCard(key, 123, "Comment");
151 
152         assertEquals("h5", key, hc.getKey());
153         assertEquals("h6", "123", hc.getValue());
154         assertEquals("h7", "Comment", hc.getComment());
155 
156         hc = new HeaderCard(card);
157         assertEquals("h8", key, hc.getKey());
158         assertEquals("h9", "123", hc.getValue());
159         assertEquals("h10", "Comment", hc.getComment());
160     }
