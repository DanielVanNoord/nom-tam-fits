38  /**
39   * Test adding a little junk after a valid image. We wish to test three
40   * scenarios: Junk at the beginning (should continue to fail) Short (<80 byte)
41   * junk after valid HDU Long (>80 byte) junk after valid HDU The last two should
42   * succeed after FitsFactory.setAllowTerminalJunk(true).
43   */
44  public class DupTest {
45  
46      @Test
47      public void test() throws Exception {
48  
49          Fits f = new Fits("src/test/resources/nom/tam/fits/test/test_dup.fits");
50          Header hdr = f.readHDU().getHeader();
51          assertEquals("Internal size:", hdr.getSize(), 2880);
52          assertEquals("External size:", hdr.getOriginalSize(), 8640);
53          assertTrue("Has duplicates:", hdr.hadDuplicates());
54          List<HeaderCard> dups = hdr.getDuplicates();
55          System.out.println("Number of duplicates:" + dups.size());
56          assertTrue("Has dups:", dups != null && dups.size() > 0);
57          assertTrue("Not rewriteable:", !hdr.rewriteable());
58          BufferedFile bf = new BufferedFile("target/created_dup.fits", "rw");
59          f.write(bf);
60          hdr.resetOriginalSize();
61          assertEquals("External size, after reset", hdr.getOriginalSize(), 2880);
62          Fits g = new Fits("target/created_dup.fits");
63          hdr = g.readHDU().getHeader();
64          assertEquals("Internal size, after rewrite", hdr.getSize(), 2880);
65          assertEquals("External size, after rewrite", hdr.getOriginalSize(), 2880);
66          assertTrue("Now rewriteable", hdr.rewriteable());
67          assertTrue("No duplicates", !hdr.hadDuplicates());
68          assertTrue("Dups is null", hdr.getDuplicates() == null);
69      }
70  }
