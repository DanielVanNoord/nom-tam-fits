37      @Test
38      public void simpleImages() throws Exception {
39          float[][] img = new float[300][300];
40  
41          Fits f = new Fits();
42  
43          ImageHDU hdu = (ImageHDU) Fits.makeHDU(img);
44          BufferedFile bf = new BufferedFile("target/ht1.fits", "rw");
45          f.addHDU(hdu);
46          f.write(bf);
47          bf.close();
48  
49          f = new Fits("target/ht1.fits");
50          hdu = (ImageHDU) f.getHDU(0);
51          Header hdr = hdu.getHeader();
52  
53          assertEquals("NAXIS", 2, hdr.getIntValue("NAXIS"));
54          assertEquals("NAXIS1", 300, hdr.getIntValue("NAXIS1"));
55          assertEquals("NAXIS2", 300, hdr.getIntValue("NAXIS2"));
56          assertEquals("NAXIS2a", 300, hdr.getIntValue("NAXIS2", -1));
57          assertEquals("NAXIS3", -1, hdr.getIntValue("NAXIS3", -1));
58  
59          assertEquals("BITPIX", -32, hdr.getIntValue("BITPIX"));
60  
61          Cursor c = hdr.iterator();
62          HeaderCard hc = (HeaderCard) c.next();
63          assertEquals("SIMPLE_1", "SIMPLE", hc.getKey());
64  
65          hc = (HeaderCard) c.next();
66          assertEquals("BITPIX_2", "BITPIX", hc.getKey());
67  
68          hc = (HeaderCard) c.next();
69          assertEquals("NAXIS_3", "NAXIS", hc.getKey());
70  
71          hc = (HeaderCard) c.next();
72          assertEquals("NAXIS1_4", "NAXIS1", hc.getKey());
73  
74          hc = (HeaderCard) c.next();
75          assertEquals("NAXIS2_5", "NAXIS2", hc.getKey());
76      }

79      @Test
80      public void extendTest() throws Exception {
81          simpleImages();
82          Fits f = new Fits("target/ht1.fits");
83          Header h = f.getHDU(0).getHeader();
84          h.addValue("TESTKEY", "TESTVAL", "TESTCOMM");
85          h.rewrite();
86          f.getStream().close();
87          f = new Fits("target/ht1.fits");
88          h = f.getHDU(0).getHeader();
89  
90          // We should be pointed after the EXTEND and before TESTKEY
91          h.addValue("TESTKEY2", "TESTVAL2", null); // Should precede TESTKEY
92  
93          Cursor c = h.iterator();
94          assertEquals("E1", ((HeaderCard) c.next()).getKey(), "SIMPLE");
95          assertEquals("E2", ((HeaderCard) c.next()).getKey(), "BITPIX");
96          assertEquals("E3", ((HeaderCard) c.next()).getKey(), "NAXIS");
97          assertEquals("E4", ((HeaderCard) c.next()).getKey(), "NAXIS1");
98          assertEquals("E5", ((HeaderCard) c.next()).getKey(), "NAXIS2");
99          assertEquals("E6", ((HeaderCard) c.next()).getKey(), "EXTEND");
100         assertEquals("E7", ((HeaderCard) c.next()).getKey(), "TESTKEY2");
101         assertEquals("E8", ((HeaderCard) c.next()).getKey(), "TESTKEY");
102

105     @Test
106     public void cursorTest() throws Exception {
107 
108         Fits f = new Fits("target/ht1.fits");
109         ImageHDU hdu = (ImageHDU) f.getHDU(0);
110         Header hdr = hdu.getHeader();
111         Cursor c = hdr.iterator();
112 
113         c.setKey("XXX");
114         c.add("CTYPE1", new HeaderCard("CTYPE1", "GLON-CAR", "Galactic Longitude"));
115         c.add("CTYPE2", new HeaderCard("CTYPE2", "GLAT-CAR", "Galactic Latitude"));
116         c.setKey("CTYPE1"); // Move before CTYPE1
117         c.add("CRVAL1", new HeaderCard("CRVAL1", 0., "Longitude at reference"));
118         c.setKey("CTYPE2"); // Move before CTYPE2
119         c.add("CRVAL2", new HeaderCard("CRVAL2", -90., "Latitude at reference"));
120         c.setKey("CTYPE1"); // Just practicing moving around!!
121         c.add("CRPIX1", new HeaderCard("CRPIX1", 150.0, "Reference Pixel X"));
122         c.setKey("CTYPE2");
123         c.add("CRPIX2", new HeaderCard("CRPIX2", 0., "Reference pixel Y"));
124         c.add("INV2", new HeaderCard("INV2", true, "Invertible axis"));
125         c.add("SYM2", new HeaderCard("SYM2", "YZ SYMMETRIC", "Symmetries..."));
126 
127         assertEquals("CTYPE1", "GLON-CAR", hdr.getStringValue("CTYPE1"));
128         assertEquals("CRPIX2", 0., hdr.getDoubleValue("CRPIX2", -2.), 0);
129 
130         c.setKey("CRVAL1");
131         HeaderCard hc = (HeaderCard) c.next();
132         assertEquals("CRVAL1_c", "CRVAL1", hc.getKey());
133         hc = (HeaderCard) c.next();
134         assertEquals("CRPIX1_c", "CRPIX1", hc.getKey());
135         hc = (HeaderCard) c.next();
136         assertEquals("CTYPE1_c", "CTYPE1", hc.getKey());
137         hc = (HeaderCard) c.next();
138         assertEquals("CRVAL2_c", "CRVAL2", hc.getKey());
139         hc = (HeaderCard) c.next();
140         assertEquals("CRPIX2_c", "CRPIX2", hc.getKey());
141         hc = (HeaderCard) c.next();
142         assertEquals("INV2_c", "INV2", hc.getKey());
143         hc = (HeaderCard) c.next();
144         assertEquals("SYM2_c", "SYM2", hc.getKey());
145         hc = (HeaderCard) c.next();
146         assertEquals("CTYPE2_c", "CTYPE2", hc.getKey());
147 
148         hdr.findCard("CRPIX1");
149         hdr.addValue("INTVAL1", 1, "An integer value");
150         hdr.addValue("LOG1", true, "A true value");
151         hdr.addValue("LOGB1", false, "A false value");
152         hdr.addValue("FLT1", 1.34, "A float value");
153         hdr.addValue("FLT2", -1.234567890e-134, "A very long float");
154         hdr.insertComment("Comment after flt2");
155 
156         c.setKey("INTVAL1");
157         hc = (HeaderCard) c.next();
158         assertEquals("INTVAL1", "INTVAL1", hc.getKey());
159         hc = (HeaderCard) c.next();
160         assertEquals("LOG1", "LOG1", hc.getKey());
161         hc = (HeaderCard) c.next();
162         assertEquals("LOGB1", "LOGB1", hc.getKey());
163         hc = (HeaderCard) c.next();
164         assertEquals("FLT1", "FLT1", hc.getKey());
165         hc = (HeaderCard) c.next();
166         assertEquals("FLT2", "FLT2", hc.getKey());
167         c.next(); // Skip comment
168         hc = (HeaderCard) c.next();
169         assertEquals("CRPIX1x", "CRPIX1", hc.getKey());
170 
171         assertEquals("FLT1", 1.34, hdr.getDoubleValue("FLT1", 0), 0);
172         c.setKey("FLT1");
173         c.next();
174         c.remove();
175         assertEquals("FLT1", 0., hdr.getDoubleValue("FLT1", 0), 0);
176         c.setKey("LOGB1");
177         hc = (HeaderCard) c.next();
178         assertEquals("AftDel1", "LOGB1", hc.getKey());
179         hc = (HeaderCard) c.next();
180         assertEquals("AftDel2", "FLT2", hc.getKey());
181         hc = (HeaderCard) c.next();
182         assertEquals("AftDel3", "Comment after flt2", hc.getComment());
183     }

206     @Test
207     public void testUpdateHeaderComments() throws Exception {
208         byte[][] z = new byte[4][4];
209         Fits f = new Fits();
210         f.addHDU(FitsFactory.HDUFactory(z));
211         BufferedFile bf = new BufferedFile("target/hx1.fits", "rw");
212         f.write(bf);
213         bf.close();
214         f = new Fits("target/hx1.fits");
215         HeaderCard c1 = f.getHDU(0).getHeader().findCard("SIMPLE");
216         assertEquals("tuhc1", c1.getComment(), HeaderCommentsMap.getComment("header:simple:1"));
217         c1 = f.getHDU(0).getHeader().findCard("BITPIX");
218         assertEquals("tuhc2", c1.getComment(), HeaderCommentsMap.getComment("header:bitpix:1"));
219         HeaderCommentsMap.updateComment("header:bitpix:1", "A byte array");
220         HeaderCommentsMap.deleteComment("header:simple:1");
221         f = new Fits();
222         f.addHDU(FitsFactory.HDUFactory(z));
223         bf = new BufferedFile("target/hx2.fits", "rw");
224         f.write(bf);
225         bf.close();
226         f = new Fits("target/hx2.fits");
227         c1 = f.getHDU(0).getHeader().findCard("SIMPLE");
228         assertEquals("tuhc1", c1.getComment(), null);
229         c1 = f.getHDU(0).getHeader().findCard("BITPIX");
230         assertEquals("tuhc2", c1.getComment(), "A byte array");
231     }

233     @Test
234     public void testRewrite() throws Exception {
235 
236         // Should be rewriteable until we add enough cards to
237         // start a new block.
238 
239         Fits f = new Fits("target/ht1.fits");
240         ImageHDU hdu = (ImageHDU) f.getHDU(0);
241         Header hdr = hdu.getHeader();
242         Cursor c = hdr.iterator();
243 
244         int nc = hdr.getNumberOfCards();
245         int nb = (nc - 1) / 36;
246 
247         while (hdr.rewriteable()) {
248             int nbx = (hdr.getNumberOfCards() - 1) / 36;
249             assertEquals("Rewrite:" + nbx, nb == nbx, hdr.rewriteable());
250             c.add(new HeaderCard("DUMMY" + nbx, null, null));
251         }
252     }

254     @Test
255     public void longStringTest() throws Exception {
256 
257         Header hdr = new Fits("target/ht1.fits").getHDU(0).getHeader();
258 
259         String seq = "0123456789";
260         String lng = "";
261         for (int i = 0; i < 20; i += 1) {
262             lng += seq;
263         }
264         assertEquals("Initial state:", false, Header.getLongStringsEnabled());
265         Header.setLongStringsEnabled(true);
266         assertEquals("Set state:", true, Header.getLongStringsEnabled());
267         hdr.addValue("LONG1", lng, "Here is a comment");
268         hdr.addValue("LONG2", "xx'yy'zz" + lng, "Another comment");
269         hdr.addValue("SHORT", "A STRING ENDING IN A &", null);
270         hdr.addValue("LONGISH", lng + "&", null);
271         hdr.addValue("LONGSTRN", "OGIP 1.0", "Uses long strings");
272 
273         String sixty = seq + seq + seq + seq + seq + seq;
274         hdr.addValue("APOS1", sixty + "''''''''''", "Should be 70 chars long");
275         hdr.addValue("APOS2", sixty + " ''''''''''", "Should be 71 chars long");
276 
277         // Now try to read the values back.
278         BufferedFile bf = new BufferedFile("target/ht4.hdr", "rw");
279         hdr.write(bf);
280         bf.close();
281         String val = hdr.getStringValue("LONG1");
282         assertEquals("LongT1", val, lng);
283         val = hdr.getStringValue("LONG2");
284         assertEquals("LongT2", val, "xx'yy'zz" + lng);
285         assertEquals("APOS1", hdr.getStringValue("APOS1").length(), 70);
286         assertEquals("APOS2", hdr.getStringValue("APOS2").length(), 71);
287         Header.setLongStringsEnabled(false);
288         val = hdr.getStringValue("LONG1");
289         assertEquals("LongT3", true, !val.equals(lng));
290         assertEquals("Longt4", true, val.length() <= 70);
291         assertEquals("longamp1", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
292         bf = new BufferedFile("target/ht4.hdr", "r");
293         hdr = new Header(bf);
294         assertEquals("Set state2:", true, Header.getLongStringsEnabled());
295         val = hdr.getStringValue("LONG1");
296         assertEquals("LongT5", val, lng);
297         val = hdr.getStringValue("LONG2");
298         assertEquals("LongT6", val, "xx'yy'zz" + lng);
299         assertEquals("longamp2", hdr.getStringValue("LONGISH"), lng + "&");
300         assertEquals("APOS1b", hdr.getStringValue("APOS1").length(), 70);
301         assertEquals("APOS2b", hdr.getStringValue("APOS2").length(), 71);
302         assertEquals("APOS2c", hdr.getStringValue("APOS2"), sixty + " ''''''''''");
303         assertEquals("longamp1b", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
304         assertEquals("longamp2b", hdr.getStringValue("LONGISH"), lng + "&");
305 
306         int cnt = hdr.getNumberOfCards();
307         // This should remove all three cards associated with
308         // LONG1
309         hdr.removeCard("LONG1");
310         assertEquals("deltest", cnt - 3, hdr.getNumberOfCards());
311         Header.setLongStringsEnabled(false);
312         // With long strings disabled this should only remove one more card.
313         hdr.removeCard("LONG2");
314         assertEquals("deltest2", cnt - 4, hdr.getNumberOfCards());
315 
316     }
