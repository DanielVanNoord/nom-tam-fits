44      @Test
45      public void simpleImages() throws Exception {
46          float[][] img = new float[300][300];
47  
48          Fits f = new Fits();
49  
50          ImageHDU hdu = (ImageHDU) Fits.makeHDU(img);
51          BufferedFile bf = new BufferedFile("target/ht1.fits", "rw");
52          f.addHDU(hdu);
53          f.write(bf);
54          bf.close();
55  
56          f = new Fits("target/ht1.fits");
57          hdu = (ImageHDU) f.getHDU(0);
58          Header hdr = hdu.getHeader();
59  
60          assertEquals("NAXIS", 2, hdr.getIntValue("NAXIS"));
61          assertEquals("NAXIS1", 300, hdr.getIntValue("NAXIS1"));
62          assertEquals("NAXIS2", 300, hdr.getIntValue("NAXIS2"));
63          assertEquals("NAXIS2a", 300, hdr.getIntValue("NAXIS2", -1));
64          assertEquals("NAXIS3", -1, hdr.getIntValue("NAXIS3", -1));
65  
66          assertEquals("BITPIX", -32, hdr.getIntValue("BITPIX"));
67  
68          Cursor c = hdr.iterator();
69          HeaderCard hc = (HeaderCard) c.next();
70          assertEquals("SIMPLE_1", "SIMPLE", hc.getKey());
71  
72          hc = (HeaderCard) c.next();
73          assertEquals("BITPIX_2", "BITPIX", hc.getKey());
74  
75          hc = (HeaderCard) c.next();
76          assertEquals("NAXIS_3", "NAXIS", hc.getKey());
77  
78          hc = (HeaderCard) c.next();
79          assertEquals("NAXIS1_4", "NAXIS1", hc.getKey());
80  
81          hc = (HeaderCard) c.next();
82          assertEquals("NAXIS2_5", "NAXIS2", hc.getKey());
83      }

86      @Test
87      public void extendTest() throws Exception {
88          simpleImages();
89          Fits f = new Fits("target/ht1.fits");
90          Header h = f.getHDU(0).getHeader();
91          h.addValue("TESTKEY", "TESTVAL", "TESTCOMM");
92          h.rewrite();
93          f.getStream().close();
94          f = new Fits("target/ht1.fits");
95          h = f.getHDU(0).getHeader();
96  
97          // We should be pointed after the EXTEND and before TESTKEY
98          h.addValue("TESTKEY2", "TESTVAL2", null); // Should precede TESTKEY
99  
100         Cursor c = h.iterator();
101         assertEquals("E1", ((HeaderCard) c.next()).getKey(), "SIMPLE");
102         assertEquals("E2", ((HeaderCard) c.next()).getKey(), "BITPIX");
103         assertEquals("E3", ((HeaderCard) c.next()).getKey(), "NAXIS");
104         assertEquals("E4", ((HeaderCard) c.next()).getKey(), "NAXIS1");
105         assertEquals("E5", ((HeaderCard) c.next()).getKey(), "NAXIS2");
106         assertEquals("E6", ((HeaderCard) c.next()).getKey(), "EXTEND");
107         assertEquals("E7", ((HeaderCard) c.next()).getKey(), "TESTKEY2");
108         assertEquals("E8", ((HeaderCard) c.next()).getKey(), "TESTKEY");
109

112     @Test
113     public void cursorTest() throws Exception {
114 
115         Fits f = new Fits("target/ht1.fits");
116         ImageHDU hdu = (ImageHDU) f.getHDU(0);
117         Header hdr = hdu.getHeader();
118         Cursor c = hdr.iterator();
119 
120         c.setKey("XXX");
121         c.add("CTYPE1", new HeaderCard("CTYPE1", "GLON-CAR", "Galactic Longitude"));
122         c.add("CTYPE2", new HeaderCard("CTYPE2", "GLAT-CAR", "Galactic Latitude"));
123         c.setKey("CTYPE1"); // Move before CTYPE1
124         c.add("CRVAL1", new HeaderCard("CRVAL1", 0., "Longitude at reference"));
125         c.setKey("CTYPE2"); // Move before CTYPE2
126         c.add("CRVAL2", new HeaderCard("CRVAL2", -90., "Latitude at reference"));
127         c.setKey("CTYPE1"); // Just practicing moving around!!
128         c.add("CRPIX1", new HeaderCard("CRPIX1", 150.0, "Reference Pixel X"));
129         c.setKey("CTYPE2");
130         c.add("CRPIX2", new HeaderCard("CRPIX2", 0., "Reference pixel Y"));
131         c.add("INV2", new HeaderCard("INV2", true, "Invertible axis"));
132         c.add("SYM2", new HeaderCard("SYM2", "YZ SYMMETRIC", "Symmetries..."));
133 
134         assertEquals("CTYPE1", "GLON-CAR", hdr.getStringValue("CTYPE1"));
135         assertEquals("CRPIX2", 0., hdr.getDoubleValue("CRPIX2", -2.), 0);
136 
137         c.setKey("CRVAL1");
138         HeaderCard hc = (HeaderCard) c.next();
139         assertEquals("CRVAL1_c", "CRVAL1", hc.getKey());
140         hc = (HeaderCard) c.next();
141         assertEquals("CRPIX1_c", "CRPIX1", hc.getKey());
142         hc = (HeaderCard) c.next();
143         assertEquals("CTYPE1_c", "CTYPE1", hc.getKey());
144         hc = (HeaderCard) c.next();
145         assertEquals("CRVAL2_c", "CRVAL2", hc.getKey());
146         hc = (HeaderCard) c.next();
147         assertEquals("CRPIX2_c", "CRPIX2", hc.getKey());
148         hc = (HeaderCard) c.next();
149         assertEquals("INV2_c", "INV2", hc.getKey());
150         hc = (HeaderCard) c.next();
151         assertEquals("SYM2_c", "SYM2", hc.getKey());
152         hc = (HeaderCard) c.next();
153         assertEquals("CTYPE2_c", "CTYPE2", hc.getKey());
154 
155         hdr.findCard("CRPIX1");
156         hdr.addValue("INTVAL1", 1, "An integer value");
157         hdr.addValue("LOG1", true, "A true value");
158         hdr.addValue("LOGB1", false, "A false value");
159         hdr.addValue("FLT1", 1.34, "A float value");
160         hdr.addValue("FLT2", -1.234567890e-134, "A very long float");
161         hdr.insertComment("Comment after flt2");
162 
163         c.setKey("INTVAL1");
164         hc = (HeaderCard) c.next();
165         assertEquals("INTVAL1", "INTVAL1", hc.getKey());
166         hc = (HeaderCard) c.next();
167         assertEquals("LOG1", "LOG1", hc.getKey());
168         hc = (HeaderCard) c.next();
169         assertEquals("LOGB1", "LOGB1", hc.getKey());
170         hc = (HeaderCard) c.next();
171         assertEquals("FLT1", "FLT1", hc.getKey());
172         hc = (HeaderCard) c.next();
173         assertEquals("FLT2", "FLT2", hc.getKey());
174         c.next(); // Skip comment
175         hc = (HeaderCard) c.next();
176         assertEquals("CRPIX1x", "CRPIX1", hc.getKey());
177 
178         assertEquals("FLT1", 1.34, hdr.getDoubleValue("FLT1", 0), 0);
179         c.setKey("FLT1");
180         c.next();
181         c.remove();
182         assertEquals("FLT1", 0., hdr.getDoubleValue("FLT1", 0), 0);
183         c.setKey("LOGB1");
184         hc = (HeaderCard) c.next();
185         assertEquals("AftDel1", "LOGB1", hc.getKey());
186         hc = (HeaderCard) c.next();
187         assertEquals("AftDel2", "FLT2", hc.getKey());
188         hc = (HeaderCard) c.next();
189         assertEquals("AftDel3", "Comment after flt2", hc.getComment());
190     }

213     @Test
214     public void testUpdateHeaderComments() throws Exception {
215         byte[][] z = new byte[4][4];
216         Fits f = new Fits();
217         f.addHDU(FitsFactory.HDUFactory(z));
218         BufferedFile bf = new BufferedFile("target/hx1.fits", "rw");
219         f.write(bf);
220         bf.close();
221         f = new Fits("target/hx1.fits");
222         HeaderCard c1 = f.getHDU(0).getHeader().findCard("SIMPLE");
223         assertEquals("tuhc1", c1.getComment(), HeaderCommentsMap.getComment("header:simple:1"));
224         c1 = f.getHDU(0).getHeader().findCard("BITPIX");
225         assertEquals("tuhc2", c1.getComment(), HeaderCommentsMap.getComment("header:bitpix:1"));
226         HeaderCommentsMap.updateComment("header:bitpix:1", "A byte array");
227         HeaderCommentsMap.deleteComment("header:simple:1");
228         f = new Fits();
229         f.addHDU(FitsFactory.HDUFactory(z));
230         bf = new BufferedFile("target/hx2.fits", "rw");
231         f.write(bf);
232         bf.close();
233         f = new Fits("target/hx2.fits");
234         c1 = f.getHDU(0).getHeader().findCard("SIMPLE");
235         assertEquals("tuhc1", c1.getComment(), null);
236         c1 = f.getHDU(0).getHeader().findCard("BITPIX");
237         assertEquals("tuhc2", c1.getComment(), "A byte array");
238     }

240     @Test
241     public void testRewrite() throws Exception {
242 
243         // Should be rewriteable until we add enough cards to
244         // start a new block.
245 
246         Fits f = new Fits("target/ht1.fits");
247         ImageHDU hdu = (ImageHDU) f.getHDU(0);
248         Header hdr = hdu.getHeader();
249         Cursor c = hdr.iterator();
250 
251         int nc = hdr.getNumberOfCards();
252         int nb = (nc - 1) / 36;
253 
254         while (hdr.rewriteable()) {
255             int nbx = (hdr.getNumberOfCards() - 1) / 36;
256             assertEquals("Rewrite:" + nbx, nb == nbx, hdr.rewriteable());
257             c.add(new HeaderCard("DUMMY" + nbx, null, null));
258         }
259     }

261     @Test
262     public void longStringTest() throws Exception {
263 
264         Header hdr = new Fits("target/ht1.fits").getHDU(0).getHeader();
265 
266         String seq = "0123456789";
267         String lng = "";
268         for (int i = 0; i < 20; i += 1) {
269             lng += seq;
270         }
271         assertEquals("Initial state:", false, Header.getLongStringsEnabled());
272         Header.setLongStringsEnabled(true);
273         assertEquals("Set state:", true, Header.getLongStringsEnabled());
274         hdr.addValue("LONG1", lng, "Here is a comment");
275         hdr.addValue("LONG2", "xx'yy'zz" + lng, "Another comment");
276         hdr.addValue("SHORT", "A STRING ENDING IN A &", null);
277         hdr.addValue("LONGISH", lng + "&", null);
278         hdr.addValue("LONGSTRN", "OGIP 1.0", "Uses long strings");
279 
280         String sixty = seq + seq + seq + seq + seq + seq;
281         hdr.addValue("APOS1", sixty + "''''''''''", "Should be 70 chars long");
282         hdr.addValue("APOS2", sixty + " ''''''''''", "Should be 71 chars long");
283 
284         // Now try to read the values back.
285         BufferedFile bf = new BufferedFile("target/ht4.hdr", "rw");
286         hdr.write(bf);
287         bf.close();
288         String val = hdr.getStringValue("LONG1");
289         assertEquals("LongT1", val, lng);
290         val = hdr.getStringValue("LONG2");
291         assertEquals("LongT2", val, "xx'yy'zz" + lng);
292         assertEquals("APOS1", hdr.getStringValue("APOS1").length(), 70);
293         assertEquals("APOS2", hdr.getStringValue("APOS2").length(), 71);
294         Header.setLongStringsEnabled(false);
295         val = hdr.getStringValue("LONG1");
296         assertEquals("LongT3", true, !val.equals(lng));
297         assertEquals("Longt4", true, val.length() <= 70);
298         assertEquals("longamp1", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
299         bf = new BufferedFile("target/ht4.hdr", "r");
300         hdr = new Header(bf);
301         assertEquals("Set state2:", true, Header.getLongStringsEnabled());
302         val = hdr.getStringValue("LONG1");
303         assertEquals("LongT5", val, lng);
304         val = hdr.getStringValue("LONG2");
305         assertEquals("LongT6", val, "xx'yy'zz" + lng);
306         assertEquals("longamp2", hdr.getStringValue("LONGISH"), lng + "&");
307         assertEquals("APOS1b", hdr.getStringValue("APOS1").length(), 70);
308         assertEquals("APOS2b", hdr.getStringValue("APOS2").length(), 71);
309         assertEquals("APOS2c", hdr.getStringValue("APOS2"), sixty + " ''''''''''");
310         assertEquals("longamp1b", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
311         assertEquals("longamp2b", hdr.getStringValue("LONGISH"), lng + "&");
312 
313         int cnt = hdr.getNumberOfCards();
314         // This should remove all three cards associated with
315         // LONG1
316         hdr.removeCard("LONG1");
317         assertEquals("deltest", cnt - 3, hdr.getNumberOfCards());
318         Header.setLongStringsEnabled(false);
319         // With long strings disabled this should only remove one more card.
320         hdr.removeCard("LONG2");
321         assertEquals("deltest2", cnt - 4, hdr.getNumberOfCards());
322 
323     }
