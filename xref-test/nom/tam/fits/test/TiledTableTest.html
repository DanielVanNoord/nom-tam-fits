105     private String createImage(String ext) throws Exception {
106         Map<String, String> params = new HashMap<String, String>();
107         params.put("compression", ext);
108         Fits f = new Fits();
109 
110         byte[][] bimg = new byte[40][40];
111         for (int i = 10; i < 30; i += 1) {
112             for (int j = 10; j < 30; j += 1) {
113                 bimg[i][j] = (byte) (i + j);
114             }
115         }
116 
117         short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);
118         int[][] iimg = (int[][]) ArrayFuncs.convertArray(bimg, int.class);
119         long[][] limg = (long[][]) ArrayFuncs.convertArray(bimg, long.class);
120         float[][] fimg = (float[][]) ArrayFuncs.convertArray(bimg, float.class);
121         double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);
122         for (int i = 0; i < 10; i += 1) {
123             for (int j = 0; j < 20; j += 1) {
124                 simg[i][j] = (short) (i + j);
125                 iimg[i][j] = i + j;
126                 limg[i][j] = i + j;
127                 fimg[i][j] = i + j;
128                 dimg[i][j] = i + j;
129             }
130         }
131 
132         // Make HDUs of various types.
133         f.addHDU(tiledImageHdu(bimg, params));
134         f.addHDU(tiledImageHdu(simg, params));
135         f.addHDU(tiledImageHdu(iimg, params));
136         f.addHDU(tiledImageHdu(fimg, params));
137         // only 32 bit supported in rice
138         if (!ext.equals("rice")) {
139             f.addHDU(tiledImageHdu(limg, params));
140             f.addHDU(tiledImageHdu(dimg, params));
141             assertEquals("HDU count before", f.getNumberOfHDUs(), 7);
142         } else {
143             assertEquals("HDU count before", f.getNumberOfHDUs(), 5);
144         }
145 
146         // Write a FITS file.
147 
148         String filename = "target/pre-tiled-image" + ext + ".fits";
149         BufferedFile bf = new BufferedFile(filename, "rw");
150         f.write(bf);
151         bf.flush();
152         bf.close();
153         return filename;
154     }
155 
156     private TiledImageHDU tiledImageHdu(Object bimg, Map<String, String> params) throws FitsException, IOException {
157         Data data = ImageHDU.encapsulate(bimg);
158         return new TiledImageHDU(new ImageHDU(ImageHDU.manufactureHeader(data), data), params);
159     }
