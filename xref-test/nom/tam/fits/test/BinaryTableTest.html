116         for (int i = 0; i < bytes.length; i += 1) {
117             bytes[i] = (byte) (2 * i);
118             bits[i][0] = bytes[i];
119             bits[i][1] = (byte) (~bytes[i]);
120             bools[i] = (bytes[i] % 8) == 0 ? true : false;
121 
122             shorts[i][0] = (short) (2 * i);
123             shorts[i][1] = (short) (3 * i);
124             shorts[i][2] = (short) (4 * i);
125 
126             ints[i] = i * i;
127             for (int j = 0; j < 4; j += 1) {
128                 for (int k = 0; k < 4; k += 1) {
129                     floats[i][j][k] = (float) (i + j * Math.exp(k));
130                 }
131             }
132             doubles[i] = 3 * Math.sin(i);
133             longs[i] = i * i * i * i;
134             strings[i] = "abcdefghijklmnopqrstuvwxzy".substring(0, i % 20);
135 
136             vf[i] = new float[i + 1];
137             vf[i][i / 2] = i * 3;
138             vs[i] = new short[i / 10 + 1];
139             vs[i][i / 10] = (short) -i;
140             vd[i] = new double[i % 2 == 0 ? 1 : 2];
141             vd[i][0] = 99.99;
142             vbool[i] = new boolean[i / 10];
143             if (i >= 10) {
144                 vbool[i][0] = i % 2 == 1;
145             }
146 
147             int m5 = i % 5;
148             vc[i] = new float[m5][];
149             for (int j = 0; j < m5; j += 1) {
150                 vc[i][j] = new float[2];
151                 vc[i][j][0] = i;
152                 vc[i][j][1] = -j;
153             }

155             for (int j = 0; j < m5; j += 1) {
156                 vdc[i][j] = new double[2];
157                 vdc[i][j][0] = -j;
158                 vdc[i][j][1] = i;

856     @Test
857     public void columnMetaTest() throws Exception {
858         Object[] data = new Object[]{
859             shorts,
860             ints,
861             floats,
862             doubles
863         };
864 
865         Fits f = new Fits();
866 
867         // Add two identical HDUs
868         BinaryTableHDU bhdu = (BinaryTableHDU) Fits.makeHDU(data);
869         f.addHDU(bhdu);
870 
871         // makeHDU creates the TFORM keywords and sometimes
872         // the TDIM keywords. Let's add some additional
873         // column metadata. For each column we'll want a TTYPE, TCOMM,
874         // TUNIT and TX and TY
875         // value and we want the final header to be in this order
876         // TTYPE, TCOMM, TFORM, [TDIM,] TUNIT, TX, TY
877         int oldNCols = bhdu.getNCols();
878 
879         for (int i = 0; i < bhdu.getNCols(); i += 1) {
880             bhdu.setColumnMeta(i, "TTYPE", "NAM" + (i + 1), null, false);
881             bhdu.setColumnMeta(i, "TCOMM", true, "Comment in comment", false);
882             bhdu.setColumnMeta(i, "TUNIT", "UNIT" + (i + 1), null, true);
883             bhdu.setColumnMeta(i, "TX", (i + 1), null, true);
884             bhdu.setColumnMeta(i, "TY", 2. * (i + 1), null, true);
885         }
886 
887         BufferedFile ff = new BufferedFile("target/bt10.fits", "rw");
888         f.write(ff);
889         ff.close();
890         f = new Fits("target/bt10.fits");
891 
892         bhdu = (BinaryTableHDU) f.getHDU(1);
893         Header hdr = bhdu.getHeader();
894         assertEquals("metaCount", oldNCols, bhdu.getNCols());
895         for (int i = 0; i < bhdu.getNCols(); i += 1) {
896             // If this worked, the first header should be the TTYPE
897             hdr.findCard("TTYPE" + (i + 1));
898             HeaderCard hc = hdr.nextCard();
899             assertEquals("M" + i + "0", "TTYPE" + (i + 1), hc.getKey());
900             hc = hdr.nextCard();
901             assertEquals("M" + i + "A", "TCOMM" + (i + 1), hc.getKey());
902             hc = hdr.nextCard();
903             assertEquals("M" + i + "B", "TFORM" + (i + 1), hc.getKey());
904             hc = hdr.nextCard();
905             // There may have been a TDIM keyword inserted automatically. Let's
906             // skip it if it was. It should only appear immediately after the
907             // TFORM keyword.
908             if (hc.getKey().startsWith("TDIM")) {
909                 hc = hdr.nextCard();
910             }
911             assertEquals("M" + i + "C", "TUNIT" + (i + 1), hc.getKey());
912             hc = hdr.nextCard();
913             assertEquals("M" + i + "D", "TX" + (i + 1), hc.getKey());
914             hc = hdr.nextCard();
915             assertEquals("M" + i + "E", "TY" + (i + 1), hc.getKey());
916         }
917     }
