32  import nom.tam.util.*;
33  import nom.tam.fits.*;
34  
35  import org.junit.Test;
36  import static org.junit.Assert.assertEquals;
37  import junit.framework.JUnit4TestAdapter;
38  
39  /**
40   * Test random groups formats in FITS data. Write and read random groups data
41   */
42  public class RandomGroupsTest {
43  
44      @Test
45      public void test() throws Exception {
46  
47          float[][] fa = new float[20][20];
48          float[] pa = new float[3];
49  
50          BufferedFile bf = new BufferedFile("target/rg1.fits", "rw");
51  
52          Object[][] data = new Object[1][2];
53          data[0][0] = pa;
54          data[0][1] = fa;
55  
56          // First lets write out the file painfully group by group.
57          BasicHDU hdu = Fits.makeHDU(data);
58          Header hdr = hdu.getHeader();
59          // Change the number of groups
60          hdr.addValue("GCOUNT", 20, "Number of groups");
61          hdr.write(bf);
62  
63          for (int i = 0; i < 20; i += 1) {
64  
65              for (int j = 0; j < pa.length; j += 1) {
66                  pa[j] = i + j;
67              }
68              for (int j = 0; j < fa.length; j += 1) {
69                  fa[j][j] = i * j;
70              }
71              // Write a group
72              bf.writeArray(data);
73          }
74  
75          byte[] padding = new byte[FitsUtil.padding(20 * ArrayFuncs.computeLSize(data))];
76          bf.write(padding);
77  
78          bf.flush();
79          bf.close();
80  
81          // Read back the data.
82          Fits f = new Fits("target/rg1.fits");
83          BasicHDU[] hdus = f.read();
84  
85          data = (Object[][]) hdus[0].getKernel();
86  
87          for (int i = 0; i < data.length; i += 1) {
88  
89              pa = (float[]) data[i][0];
90              fa = (float[][]) data[i][1];
91              for (int j = 0; j < pa.length; j += 1) {
92                  assertEquals("paramTest:" + i + " " + j, (float) (i + j), pa[j], 0);
93              }
94              for (int j = 0; j < fa.length; j += 1) {
95                  assertEquals("dataTest:" + i + " " + j, (float) (i * j), fa[j][j], 0);
96              }
97          }
98  
99          // Now do it in one fell swoop -- but we have to have
100         // all the data in place first.
101         f = new Fits();
102 
103         // Generate a FITS HDU from the kernel.
104         f.addHDU(Fits.makeHDU(data));
105         bf = new BufferedFile("target/rg2.fits", "rw");
106         f.write(bf);
107 
108         bf.flush();
109         bf.close();
110 
111         f = new Fits("target/rg2.fits");
112         data = (Object[][]) f.read()[0].getKernel();
113         for (int i = 0; i < data.length; i += 1) {
114 
115             pa = (float[]) data[i][0];
116             fa = (float[][]) data[i][1];
117             for (int j = 0; j < pa.length; j += 1) {
118                 assertEquals("paramTest:" + i + " " + j, (float) (i + j), pa[j], 0);
119             }
120             for (int j = 0; j < fa.length; j += 1) {
121                 assertEquals("dataTest:" + i + " " + j, (float) (i * j), fa[j][j], 0);
122             }
123         }
124     }
125 }
