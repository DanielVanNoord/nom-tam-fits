45  /**
46   * Test the ImageHDU, ImageData and ImageTiler classes. - multiple HDU's in a
47   * single file - deferred input of HDUs - creating and reading arrays of all
48   * permitted types. - Tiles of 1, 2 and 3 dimensions - from a file - from
49   * internal data - Multiple tiles extracted from an image.
50   */
51  public class ImageTest {
52  
53      @Test
54      public void test() throws Exception {
55  
56          Fits f = new Fits();
57  
58          byte[][] bimg = new byte[40][40];
59          for (int i = 10; i < 30; i += 1) {
60              for (int j = 10; j < 30; j += 1) {
61                  bimg[i][j] = (byte) (i + j);
62              }
63          }
64  
65          short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);
66          int[][] iimg = (int[][]) ArrayFuncs.convertArray(bimg, int.class);
67          long[][] limg = (long[][]) ArrayFuncs.convertArray(bimg, long.class);
68          float[][] fimg = (float[][]) ArrayFuncs.convertArray(bimg, float.class);
69          double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);
70          int[][][] img3 = new int[10][20][30];
71          for (int i = 0; i < 10; i += 1) {
72              for (int j = 0; j < 20; j += 1) {
73                  for (int k = 0; k < 30; k += 1) {
74                      img3[i][j][k] = i + j + k;
75                  }
76              }
77          }
78  
79          double[] img1 = (double[]) ArrayFuncs.flatten(dimg);
80  
81          // Make HDUs of various types.
82          f.addHDU(Fits.makeHDU(bimg));
83          f.addHDU(Fits.makeHDU(simg));
84          f.addHDU(Fits.makeHDU(iimg));
85          f.addHDU(Fits.makeHDU(limg));
86          f.addHDU(Fits.makeHDU(fimg));
87          f.addHDU(Fits.makeHDU(dimg));
88          f.addHDU(Fits.makeHDU(img3));
89          f.addHDU(Fits.makeHDU(img1));
90  
91          assertEquals("HDU count before", f.getNumberOfHDUs(), 8);
92  
93          // Write a FITS file.
94  
95          BufferedFile bf = new BufferedFile("target/image1.fits", "rw");
96          f.write(bf);
97          bf.flush();
98          bf.close();
99          bf = null;
100 
101         f = null;
102 
103         bf = new BufferedFile("target/image1.fits");
104 
105         // Read a FITS file
106         f = new Fits("target/image1.fits");
107         BasicHDU[] hdus = f.read();
108 
109         assertEquals("HDU count after", 8, f.getNumberOfHDUs());
110         assertEquals("byte image", true, ArrayFuncs.arrayEquals(bimg, hdus[0].getData().getKernel()));
111         assertEquals("short image", true, ArrayFuncs.arrayEquals(simg, hdus[1].getData().getKernel()));
112         assertEquals("int image", true, ArrayFuncs.arrayEquals(iimg, hdus[2].getData().getKernel()));
113         assertEquals("long image", true, ArrayFuncs.arrayEquals(limg, hdus[3].getData().getKernel()));
114         assertEquals("float image", true, ArrayFuncs.arrayEquals(fimg, hdus[4].getData().getKernel()));
115         assertEquals("double image", true, ArrayFuncs.arrayEquals(dimg, hdus[5].getData().getKernel()));
116         assertEquals("int3 image", true, ArrayFuncs.arrayEquals(img3, hdus[6].getData().getKernel()));
117         assertEquals("double1 image", true, ArrayFuncs.arrayEquals(img1, hdus[7].getData().getKernel()));
118     }
